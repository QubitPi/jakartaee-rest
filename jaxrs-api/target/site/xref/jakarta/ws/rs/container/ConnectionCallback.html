17  package jakarta.ws.rs.container;
18  
19  /**
20   * Asynchronous request processing lifecycle callback that receives connection related {@link AsyncResponse asynchronous
21   * response} lifecycle events.
22   * <p>
23   * Support for this type of callback by JAX-RS runtime is OPTIONAL.
24   * </p>
25   *
26   * @author Marek Potociar
27   * @since 2.0
28   */
29  public interface ConnectionCallback {
30      /**
31       * This callback notification method is invoked in case the container detects that the remote client connection
32       * associated with the asynchronous response has been disconnected.
33       *
34       * @param disconnected asynchronous response for which the remote client connection has been lost.
35       */
36      public void onDisconnect(AsyncResponse disconnected);
37  }
