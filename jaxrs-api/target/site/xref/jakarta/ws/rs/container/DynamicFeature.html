17  package jakarta.ws.rs.container;
18  
19  import jakarta.ws.rs.core.FeatureContext;
20  import jakarta.ws.rs.ext.ReaderInterceptor;
21  import jakarta.ws.rs.ext.WriterInterceptor;
22  
23  /**
24   * A JAX-RS meta-provider for dynamic registration of <i>post-matching</i> providers during a JAX-RS application setup
25   * at deployment time.
26   *
27   * Dynamic feature is used by JAX-RS runtime to register providers that shall be applied to a particular resource class
28   * and method and overrides any annotation-based binding definitions defined on any registered resource filter or
29   * interceptor instance.
30   * <p>
31   * Providers implementing this interface MAY be annotated with {@link jakarta.ws.rs.ext.Provider &#64;Provider} annotation
32   * in order to be discovered by JAX-RS runtime when scanning for resources and providers. This provider types is
33   * supported only as part of the Server API.
34   * </p>
35   *
36   * @author Santiago Pericas-Geertsen
37   * @author Bill Burke
38   * @author Marek Potociar
39   * @see jakarta.ws.rs.NameBinding
40   * @since 2.0
41   */
42  public interface DynamicFeature {
43  
44      /**
45       * A callback method called by the JAX-RS runtime during the application deployment to register provider instances or
46       * classes in a {@link jakarta.ws.rs.core.Configuration runtime configuration} scope of a particular
47       * {@link jakarta.ws.rs.HttpMethod resource or sub-resource method}; i.e. the providers that should be dynamically bound
48       * to the method.
49       * <p>
50       * The registered provider instances or classes are expected to be implementing one or more of the following interfaces:
51       * </p>
52       * <ul>
53       * <li>{@link ContainerRequestFilter}</li>
54       * <li>{@link ContainerResponseFilter}</li>
55       * <li>{@link ReaderInterceptor}</li>
56       * <li>{@link WriterInterceptor}</li>
57       * <li>{@link jakarta.ws.rs.core.Feature}</li>
58       * </ul>
59       * <p>
60       * A provider instance or class that does not implement any of the interfaces above may be ignored by the JAX-RS
61       * implementation. In such case a {@link java.util.logging.Level#WARNING warning} message must be logged. JAX-RS
62       * implementations may support additional provider contracts that can be registered using a dynamic feature concept.
63       * </p>
64       * <p>
65       * Conceptually, this callback method is called during a {@link jakarta.ws.rs.HttpMethod resource or sub-resource method}
66       * discovery phase (typically once per each discovered resource or sub-resource method) to register provider instances
67       * or classes in a {@code configuration} scope of each particular method identified by the supplied {@link ResourceInfo
68       * resource information}. The responsibility of the feature is to properly update the supplied {@code configuration}
69       * context.
70       * </p>
71       *
72       * @param resourceInfo resource class and method information.
73       * @param context configurable resource or sub-resource method-level runtime context associated with the
74       * {@code resourceInfo} in which the feature
75       */
76      public void configure(ResourceInfo resourceInfo, FeatureContext context);
77  }
