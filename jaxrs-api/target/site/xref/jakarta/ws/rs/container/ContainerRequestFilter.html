17  package jakarta.ws.rs.container;
18  
19  import java.io.IOException;
20  
21  /**
22   * An extension interface implemented by container request filters.
23   * <p>
24   * By default, i.e. if no {@link jakarta.ws.rs.NameBinding name binding} is applied to the filter implementation class,
25   * the filter instance is applied globally, however only after the incoming request has been matched to a particular
26   * resource by JAX-RS runtime. If there is a {@link jakarta.ws.rs.NameBinding &#64;NameBinding} annotation applied to the
27   * filter, the filter will also be executed at the <i>post-match</i> request extension point, but only in case the
28   * matched {@link jakarta.ws.rs.HttpMethod resource or sub-resource method} is bound to the same name-binding annotation.
29   * </p>
30   * <p>
31   * In case the filter should be applied at the <i>pre-match</i> extension point, i.e. before any request matching has
32   * been performed by JAX-RS runtime, the filter MUST be annotated with a {@link PreMatching &#64;PreMatching}
33   * annotation.
34   * </p>
35   * <p>
36   * Use a pre-match request filter to update the input to the JAX-RS matching algorithm, e.g., the HTTP method, Accept
37   * header, return cached responses etc. Otherwise, the use of a request filter invoked at the <i>post-match</i> request
38   * extension point (after a successful resource method matching) is recommended.
39   * </p>
40   * <p>
41   * Filters implementing this interface must be annotated with {@link jakarta.ws.rs.ext.Provider &#64;Provider} to be
42   * discovered by the JAX-RS runtime. Container request filter instances may also be discovered and bound
43   * {@link DynamicFeature dynamically} to particular resource methods.
44   * </p>
45   *
46   * @author Marek Potociar
47   * @author Santiago Pericas-Geertsen
48   * @see PreMatching
49   * @see jakarta.ws.rs.container.ContainerResponseFilter
50   * @since 2.0
51   */
52  public interface ContainerRequestFilter {
53  
54      /**
55       * Filter method called before a request has been dispatched to a resource.
56       *
57       * <p>
58       * Filters in the filter chain are ordered according to their {@code jakarta.annotation.Priority} class-level annotation
59       * value. If a request filter produces a response by calling {@link ContainerRequestContext#abortWith} method, the
60       * execution of the (either pre-match or post-match) request filter chain is stopped and the response is passed to the
61       * corresponding response filter chain (either pre-match or post-match). For example, a pre-match caching filter may
62       * produce a response in this way, which would effectively skip any post-match request filters as well as post-match
63       * response filters. Note however that a responses produced in this manner would still be processed by the pre-match
64       * response filter chain.
65       * </p>
66       *
67       * @param requestContext request context.
68       * @throws IOException if an I/O exception occurs.
69       * @see PreMatching
70       */
71      public void filter(ContainerRequestContext requestContext) throws IOException;
72  }
