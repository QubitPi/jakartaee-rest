17  package jakarta.ws.rs.container;
18  
19  import java.io.IOException;
20  
21  /**
22   * An extension interface implemented by container response filters.
23   * <p>
24   * By default, i.e. if no {@link jakarta.ws.rs.NameBinding name binding} is applied to the filter implementation class,
25   * the filter instance is applied globally to any outgoing response. If there is a {@code @NameBinding} annotation
26   * applied to the filter, the filter will only be executed for a response for which the request has been matched to a
27   * {@link jakarta.ws.rs.HttpMethod resource or sub-resource method} AND the method or the whole custom
28   * {@link jakarta.ws.rs.core.Application JAX-RS Application} class is bound to the same name-binding annotation.
29   * </p>
30   * <p>
31   * Implement a name-bound response filter in cases when you want limit the filter functionality to a matched resource or
32   * resource method. In other cases, when the filter should be applied globally to any outgoing response, implement an
33   * unbound, global response filter.
34   * </p>
35   * <p>
36   * Filters implementing this interface must be annotated with {@link jakarta.ws.rs.ext.Provider &#64;Provider} to be
37   * discovered by the JAX-RS runtime. Container response filter instances may also be discovered and bound
38   * {@link DynamicFeature dynamically} to particular resource methods.
39   * </p>
40   *
41   * @author Marek Potociar
42   * @author Santiago Pericas-Geertsen
43   * @see jakarta.ws.rs.container.ContainerRequestFilter
44   * @since 2.0
45   */
46  public interface ContainerResponseFilter {
47  
48      /**
49       * Filter method called after a response has been provided for a request (either by a {@link ContainerRequestFilter
50       * request filter} or by a matched resource method.
51       * <p>
52       * Filters in the filter chain are ordered according to their {@code jakarta.annotation.Priority} class-level annotation
53       * value.
54       * </p>
55       *
56       * @param requestContext request context.
57       * @param responseContext response context.
58       * @throws IOException if an I/O exception occurs.
59       */
60      public void filter(ContainerRequestContext requestContext, ContainerResponseContext responseContext)
61              throws IOException;
62  }
