17  package jakarta.ws.rs.ext;
18  
19  import java.lang.annotation.Annotation;
20  import java.lang.reflect.Type;
21  
22  import jakarta.ws.rs.core.MediaType;
23  
24  /**
25   * An injectable interface providing runtime lookup of provider instances.
26   *
27   * @author Paul Sandoz
28   * @author Marc Hadley
29   * @see jakarta.ws.rs.core.Context
30   * @see MessageBodyReader
31   * @see MessageBodyWriter
32   * @see ContextResolver
33   * @see ExceptionMapper
34   * @since 1.0
35   */
36  public interface Providers {
37  
38      /**
39       * Get a message body reader that matches a set of criteria. The set of readers is first filtered by comparing the
40       * supplied value of {@code mediaType} with the value of each reader's {@link jakarta.ws.rs.Consumes}, ensuring the
41       * supplied value of {@code type} is assignable to the generic type of the reader, and eliminating those that do not
42       * match. The list of matching readers is then ordered with those with the best matching values of
43       * {@link jakarta.ws.rs.Consumes} (x/y &gt; x&#47;* &gt; *&#47;*) sorted first. Finally, the
44       * {@link MessageBodyReader#isReadable(Class, Type, Annotation[], MediaType)} method is called on each reader in order
45       * using the supplied criteria and the first reader that returns {@code true} is selected and returned.
46       *
47       * @param <T> type of the the object that is to be read.
48       * @param type the class of the object that is to be read.
49       * @param genericType the type of object to be produced. E.g. if the message body is to be converted into a method
50       * parameter, this will be the formal type of the method parameter as returned by
51       * {@code Class.getGenericParameterTypes}.
52       * @param annotations an array of the annotations on the declaration of the artifact that will be initialized with the
53       * produced instance. E.g. if the message body is to be converted into a method parameter, this will be the annotations
54       * on that parameter returned by {@code Class.getParameterAnnotations}.
55       * @param mediaType the media type of the data that will be read.
56       * @return a MessageBodyReader that matches the supplied criteria or {@code null} if none is found.
57       */
58      <T> MessageBodyReader<T> getMessageBodyReader(Class<T> type,
59              Type genericType, Annotation[] annotations, MediaType mediaType);
60  
61      /**
62       * Get a message body writer that matches a set of criteria. The set of writers is first filtered by comparing the
63       * supplied value of {@code mediaType} with the value of each writer's {@link jakarta.ws.rs.Produces}, ensuring the
64       * supplied value of {@code type} is assignable to the generic type of the reader, and eliminating those that do not
65       * match. The list of matching writers is then ordered with those with the best matching values of
66       * {@link jakarta.ws.rs.Produces} (x/y &gt; x&#47;* &gt; *&#47;*) sorted first. Finally, the
67       * {@link MessageBodyWriter#isWriteable(Class, Type, Annotation[], MediaType)} method is called on each writer in order
68       * using the supplied criteria and the first writer that returns {@code true} is selected and returned.
69       *
70       * @param <T> type of the object that is to be written.
71       * @param type the class of the object that is to be written.
72       * @param genericType the type of object to be written. E.g. if the message body is to be produced from a field, this
73       * will be the declared type of the field as returned by {@code Field.getGenericType}.
74       * @param annotations an array of the annotations on the declaration of the artifact that will be written. E.g. if the
75       * message body is to be produced from a field, this will be the annotations on that field returned by
76       * {@code Field.getDeclaredAnnotations}.
77       * @param mediaType the media type of the data that will be written.
78       * @return a MessageBodyReader that matches the supplied criteria or {@code null} if none is found.
79       */
80      <T> MessageBodyWriter<T> getMessageBodyWriter(Class<T> type,
81              Type genericType, Annotation[] annotations, MediaType mediaType);
82  
83      /**
84       * Get an exception mapping provider for a particular class of exception. Returns the provider whose generic type is the
85       * nearest superclass of {@code type}.
86       *
87       * @param <T> type of the exception handled by the exception mapping provider.
88       * @param type the class of exception.
89       * @return an {@link ExceptionMapper} for the supplied type or {@code null} if none is found.
90       */
91      <T extends Throwable> ExceptionMapper<T> getExceptionMapper(Class<T> type);
92  
93      /**
94       * Get a context resolver for a particular type of context and media type. The set of resolvers is first filtered by
95       * comparing the supplied value of {@code mediaType} with the value of each resolver's {@link jakarta.ws.rs.Produces},
96       * ensuring the generic type of the context resolver is assignable to the supplied value of {@code contextType}, and
97       * eliminating those that do not match. If only one resolver matches the criteria then it is returned. If more than one
98       * resolver matches then the list of matching resolvers is ordered with those with the best matching values of
99       * {@link jakarta.ws.rs.Produces} (x/y &gt; x&#47;* &gt; *&#47;*) sorted first. A proxy is returned that delegates calls
100      * to {@link ContextResolver#getContext(java.lang.Class)} to each matching context resolver in order and returns the
101      * first non-null value it obtains or null if all matching context resolvers return null.
102      *
103      * @param <T> type of the context.
104      * @param contextType the class of context desired.
105      * @param mediaType the media type of data for which a context is required.
106      * @return a matching context resolver instance or {@code null} if no matching context providers are found.
107      */
108     <T> ContextResolver<T> getContextResolver(Class<T> contextType,
109             MediaType mediaType);
110 }
