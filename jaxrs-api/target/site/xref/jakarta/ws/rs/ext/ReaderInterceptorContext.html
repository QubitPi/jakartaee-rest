17  package jakarta.ws.rs.ext;
18  
19  import java.io.IOException;
20  import java.io.InputStream;
21  
22  import jakarta.ws.rs.WebApplicationException;
23  import jakarta.ws.rs.core.MultivaluedMap;
24  
25  /**
26   * Context class used by {@link jakarta.ws.rs.ext.ReaderInterceptor} to intercept calls to (@link
27   * jakarta.ws.rs.ext.MessageBodyReader#readFrom}. The getters and setters in this context class correspond to the
28   * parameters of the intercepted method.
29   *
30   * @author Santiago Pericas-Geertsen
31   * @author Bill Burke
32   * @see ReaderInterceptor
33   * @see MessageBodyReader
34   * @since 2.0
35   */
36  public interface ReaderInterceptorContext extends InterceptorContext {
37  
38      /**
39       * Proceed to the next interceptor in the chain. Return the result of the next interceptor invoked. Interceptors MUST
40       * explicitly call this method to continue the execution chain; the call to this method in the last interceptor of the
41       * chain will invoke the wrapped {@link jakarta.ws.rs.ext.MessageBodyReader#readFrom}.
42       *
43       * @return result of next interceptor invoked.
44       * @throws IOException if an IO error arises or is thrown by the wrapped {@code MessageBodyReader.readFrom} method.
45       * @throws jakarta.ws.rs.WebApplicationException thrown by the wrapped {@code MessageBodyReader.readFrom} method.
46       */
47      public Object proceed() throws IOException, WebApplicationException;
48  
49      /**
50       * Get the input stream of the object to be read. The JAX-RS runtime is responsible for closing the input stream.
51       *
52       * @return input stream of the object to be read.
53       */
54      public InputStream getInputStream();
55  
56      /**
57       * Set the input stream of the object to be read. For example, by wrapping it with another input stream. The JAX-RS
58       * runtime is responsible for closing the input stream that is set.
59       *
60       * @param is new input stream.
61       */
62      public void setInputStream(InputStream is);
63  
64      /**
65       * Get mutable map of HTTP headers.
66       * <p>
67       * Note that while the headers are mutable, a {@link ReaderInterceptor reader interceptor} should typically roll-back
68       * any header modifications once the call to {@link #proceed() context.proceed()} returns, to avoid externally visible
69       * side-effects of the interceptor invocation.
70       * </p>
71       *
72       * @return map of HTTP headers.
73       */
74      public MultivaluedMap<String, String> getHeaders();
75  }
