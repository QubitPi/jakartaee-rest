17  package jakarta.ws.rs.ext;
18  
19  /**
20   * Interface for message body reader interceptors that wrap around calls to
21   * {@link jakarta.ws.rs.ext.MessageBodyReader#readFrom}.
22   * <p>
23   * Providers implementing {@code ReaderInterceptor} contract must be either programmatically registered in a JAX-RS
24   * runtime or must be annotated with {@link jakarta.ws.rs.ext.Provider &#64;Provider} annotation to be automatically
25   * discovered by the JAX-RS runtime during a provider scanning phase. Message body interceptor instances may also be
26   * discovered and bound {@link jakarta.ws.rs.container.DynamicFeature dynamically} to particular resource methods.
27   * </p>
28   *
29   * @author Santiago Pericas-Geertsen
30   * @author Bill Burke
31   * @author Marek Potociar
32   * @see MessageBodyReader
33   * @since 2.0
34   */
35  public interface ReaderInterceptor {
36  
37      /**
38       * Interceptor method wrapping calls to {@link MessageBodyReader#readFrom} method.
39       *
40       * The parameters of the wrapped method called are available from {@code context}. Implementations of this method SHOULD
41       * explicitly call {@link ReaderInterceptorContext#proceed} to invoke the next interceptor in the chain, and ultimately
42       * the wrapped {@link MessageBodyReader#readFrom} method.
43       *
44       * @param context invocation context.
45       * @return result of next interceptor invoked or the wrapped method if last interceptor in chain.
46       * @throws java.io.IOException if an IO error arises or is thrown by the wrapped {@code MessageBodyReader.readFrom}
47       * method.
48       * @throws jakarta.ws.rs.WebApplicationException thrown by the wrapped {@code MessageBodyReader.readFrom} method.
49       */
50      public Object aroundReadFrom(ReaderInterceptorContext context)
51              throws java.io.IOException, jakarta.ws.rs.WebApplicationException;
52  }
