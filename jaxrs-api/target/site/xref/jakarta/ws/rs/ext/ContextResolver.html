17  package jakarta.ws.rs.ext;
18  
19  /**
20   * Contract for a provider that supplies context information to resource classes and other providers.
21   *
22   * A {@code ContextResolver} implementation may be annotated with {@link jakarta.ws.rs.Produces} to restrict the media
23   * types for which it will be considered suitable.
24   * <p>
25   * Providers implementing {@code ContextResolver} contract must be either programmatically registered in a JAX-RS
26   * runtime or must be annotated with {@link jakarta.ws.rs.ext.Provider &#64;Provider} annotation to be automatically
27   * discovered by the JAX-RS runtime during a provider scanning phase.
28   * </p>
29   *
30   * @param <T> type of the context
31   * @author Paul Sandoz
32   * @author Marc Hadley
33   * @see jakarta.ws.rs.core.Context
34   * @see Providers#getContextResolver(Class, jakarta.ws.rs.core.MediaType)
35   * @see Provider
36   * @see jakarta.ws.rs.Produces
37   * @since 1.0
38   */
39  public interface ContextResolver<T> {
40  
41      /**
42       * Get a context of type {@code T} that is applicable to the supplied type.
43       *
44       * @param type the class of object for which a context is desired
45       * @return a context for the supplied type or {@code null} if a context for the supplied type is not available from this
46       * provider.
47       */
48      T getContext(Class<?> type);
49  }
