17  package jakarta.ws.rs.ext;
18  
19  import java.lang.annotation.Annotation;
20  import java.lang.reflect.Type;
21  import java.util.Collection;
22  
23  import jakarta.ws.rs.core.MediaType;
24  
25  /**
26   * Context shared by message body interceptors that can be used to wrap calls to
27   * {@link jakarta.ws.rs.ext.MessageBodyReader#readFrom} and {@link jakarta.ws.rs.ext.MessageBodyWriter#writeTo}. The getters
28   * and setters in this context class correspond to the parameters in the aforementioned methods.
29   *
30   * @author Santiago Pericas-Geertsen
31   * @author Bill Burke
32   * @see ReaderInterceptor
33   * @see WriterInterceptor
34   * @see ReaderInterceptorContext
35   * @see WriterInterceptorContext
36   * @since 2.0
37   */
38  public interface InterceptorContext {
39  
40      /**
41       * Returns the property with the given name registered in the current request/response exchange context, or {@code null}
42       * if there is no property by that name.
43       * <p>
44       * A property allows a JAX-RS filters and interceptors to exchange additional custom information not already provided by
45       * this interface.
46       * </p>
47       * <p>
48       * A list of supported properties can be retrieved using {@link #getPropertyNames()}. Custom property names should
49       * follow the same convention as package names.
50       * </p>
51       * <p>
52       * In a Servlet container, on the server side, the properties are backed by the {@code ServletRequest} and contain all
53       * the attributes available in the {@code ServletRequest}.
54       * </p>
55       *
56       * @param name a {@code String} specifying the name of the property.
57       * @return an {@code Object} containing the value of the property, or {@code null} if no property exists matching the
58       * given name.
59       * @see #getPropertyNames()
60       */
61      public Object getProperty(String name);
62  
63      /**
64       * Returns {@code true} if the property with the given name is registered in the current request/response exchange
65       * context, or {@code false} if there is no property by that name.
66       * <p>
67       * Use the {@link #getProperty} method with a property name to get the value of a property.
68       * </p>
69       *
70       * @param name a {@code String} specifying the name of the property.
71       * @return {@code true} if this property is registered in the context, or {@code false} if no property exists matching
72       * the given name.
73       * @see #getPropertyNames()
74       */
75      public default boolean hasProperty(String name) {
76          return getProperty(name) != null;
77      }
78  
79      /**
80       * Returns an immutable {@link java.util.Collection collection} containing the property names available within the
81       * context of the current request/response exchange context.
82       * <p>
83       * Use the {@link #getProperty} method with a property name to get the value of a property.
84       * </p>
85       * <p>
86       * In a Servlet container, the properties are synchronized with the {@code ServletRequest} and expose all the attributes
87       * available in the {@code ServletRequest}. Any modifications of the properties are also reflected in the set of
88       * properties of the associated {@code ServletRequest}.
89       * </p>
90       *
91       * @return an immutable {@link java.util.Collection collection} of property names.
92       * @see #getProperty
93       */
94      public Collection<String> getPropertyNames();
95  
96      /**
97       * Binds an object to a given property name in the current request/response exchange context. If the name specified is
98       * already used for a property, this method will replace the value of the property with the new value.
99       * <p>
100      * A property allows a JAX-RS filters and interceptors to exchange additional custom information not already provided by
101      * this interface.
102      * </p>
103      * <p>
104      * A list of supported properties can be retrieved using {@link #getPropertyNames()}. Custom property names should
105      * follow the same convention as package names.
106      * </p>
107      * <p>
108      * If a {@code null} value is passed, the effect is the same as calling the {@link #removeProperty(String)} method.
109      * </p>
110      * <p>
111      * In a Servlet container, on the server side, the properties are backed by the {@code ServletRequest} and contain all
112      * the attributes available in the {@code ServletRequest}.
113      * </p>
114      *
115      * @param name a {@code String} specifying the name of the property.
116      * @param object an {@code Object} representing the property to be bound.
117      */
118     public void setProperty(String name, Object object);
119 
120     /**
121      * Removes a property with the given name from the current request/response exchange context. After removal, subsequent
122      * calls to {@link #getProperty} to retrieve the property value will return {@code null}.
123      * <p>
124      * In a Servlet container, on the server side, the properties are backed by the {@code ServletRequest} and contain all
125      * the attributes available in the {@code ServletRequest}.
126      * </p>
127      *
128      * @param name a {@code String} specifying the name of the property to be removed.
129      */
130     public void removeProperty(String name);
131 
132     /**
133      * Get an array of the annotations formally declared on the artifact that initiated the intercepted entity provider
134      * invocation.
135      *
136      * E.g. if the message body is to be converted into a method parameter, this will be the annotations on that parameter
137      * returned by {@link java.lang.reflect.Method#getParameterAnnotations Method.getParameterAnnotations()}; if the
138      * server-side response entity instance is to be converted into an output stream, this will be the annotations on the
139      * matched resource method returned by {@link java.lang.reflect.Method#getAnnotations() Method.getAnnotations()}.
140      *
141      * This method may return an empty array in case the interceptor is not invoked in a context of any particular resource
142      * method (e.g. as part of the client API), but will never return {@code null}.
143      *
144      * @return annotations declared on the artifact that initiated the intercepted entity provider invocation.
145      */
146     public Annotation[] getAnnotations();
147 
148     /**
149      * Update annotations on the formal declaration of the artifact that initiated the intercepted entity provider
150      * invocation.
151      *
152      * Calling this method has no effect in the client API.
153      *
154      * @param annotations updated annotations declarataion of the artifact that initiated the intercepted entity provider
155      * invocation. Must not be {@code null}.
156      * @throws NullPointerException in case the input parameter is {@code null}.
157      */
158     public void setAnnotations(Annotation[] annotations);
159 
160     /**
161      * Get Java type supported by corresponding message body provider.
162      *
163      * @return java type supported by provider
164      */
165     Class<?> getType();
166 
167     /**
168      * Update Java type before calling message body provider.
169      *
170      * @param type java type for provider
171      */
172     public void setType(Class<?> type);
173 
174     /**
175      * Get the type of the object to be produced or written.
176      *
177      * @return type of object produced or written
178      */
179     Type getGenericType();
180 
181     /**
182      * Update type of the object to be produced or written.
183      *
184      * @param genericType new type for object
185      */
186     public void setGenericType(Type genericType);
187 
188     /**
189      * Get media type of HTTP entity.
190      *
191      * @return media type of HTTP entity
192      */
193     public MediaType getMediaType();
194 
195     /**
196      * Update media type of HTTP entity.
197      *
198      * @param mediaType new type for HTTP entity
199      */
200     public void setMediaType(MediaType mediaType);
201 }
