17  package jakarta.ws.rs.ext;
18  
19  import java.lang.annotation.Documented;
20  import java.lang.annotation.ElementType;
21  import java.lang.annotation.Retention;
22  import java.lang.annotation.RetentionPolicy;
23  import java.lang.annotation.Target;
24  
25  import jakarta.ws.rs.DefaultValue;
26  
27  /**
28   * Defines a contract for a delegate responsible for converting between a {@code String} form of a message parameter
29   * value and the corresponding custom Java type {@code T}.
30   *
31   * Conversion of message parameter values injected via {@link jakarta.ws.rs.PathParam &#64;PathParam},
32   * {@link jakarta.ws.rs.QueryParam &#64;QueryParam}, {@link jakarta.ws.rs.MatrixParam &#64;MatrixParam},
33   * {@link jakarta.ws.rs.FormParam &#64;FormParam}, {@link jakarta.ws.rs.CookieParam &#64;CookieParam} and
34   * {@link jakarta.ws.rs.HeaderParam &#64;HeaderParam} is supported.
35   * JAX-RS implementations MUST support the {@code ParamConverter} mechanism for all Java types. If a
36   * {@code ParamConverter} is available for a type, it MUST be preferred over all other conversion strategies mentioned
37   * in section 3.2 (i.e. single {@code String} argument constructor, static {@code valueOf} or {@code fromString}
38   * methods, etc.).
39   * <p>
40   * By default, when used for injection of parameter values, a selected {@code ParamConverter} instance MUST be used
41   * eagerly by a JAX-RS runtime to convert any {@link DefaultValue default value} in the resource or provider model, that
42   * is during the application deployment, before any value &ndash; default or otherwise &ndash; is actually required.
43   * This conversion strategy ensures that any errors in the default values are reported as early as possible. This
44   * default behavior may be overridden by annotating the {@code ParamConverter} implementation class with a {@link Lazy
45   * &#64;Lazy} annotation. In such case any default value conversion delegated to the {@code @Lazy}-annotated converter
46   * will be deferred to a latest possible moment (i.e. until the injection of such default value is required).
47   * </p>
48   * <p>
49   * NOTE: A service implementing this contract is not recognized as a registrable JAX-RS extension provider. Instead, a
50   * {@link ParamConverterProvider} instance responsible for providing {@code ParamConverter} instances has to be
51   * registered as one of the JAX-RS extension providers.
52   * </p>
53   *
54   * @param <T> the supported Java type convertible to/from a {@code String} format.
55   * @author Marek Potociar
56   * @since 2.0
57   */
58  public interface ParamConverter<T> {
59  
60      /**
61       * Mandates that a conversion of any {@link DefaultValue default value} delegated to a {@link ParamConverter parameter
62       * converter} annotated with {@code @Lazy} annotation SHOULD occur only once the value is actually required (e.g. to be
63       * injected for the first time).
64       *
65       * @since 2.0
66       */
67      @Target({ ElementType.TYPE })
68      @Retention(RetentionPolicy.RUNTIME)
69      @Documented
70      public static @interface Lazy {
71      }
72  
73      /**
74       * Parse the supplied value and create an instance of {@code T}.
75       *
76       * @param value the string value.
77       * @return the newly created instance of {@code T}.
78       *
79       * @throws IllegalArgumentException if the supplied string cannot be parsed or is {@code null}.
80       */
81      public T fromString(String value);
82  
83      /**
84       * Convert the supplied value to a String.
85       * <p>
86       * This method is reserved for future use. Proprietary JAX-RS extensions may leverage the method. Users should be aware
87       * that any such support for the method comes at the expense of producing non-portable code.
88       * </p>
89       *
90       * @param value the value of type {@code T}.
91       * @return a String representation of the value.
92       *
93       * @throws IllegalArgumentException if the supplied object cannot be serialized or is {@code null}.
94       */
95      public String toString(T value);
96  }
