17  package jakarta.ws.rs.ext;
18  
19  import java.lang.annotation.Annotation;
20  import java.lang.reflect.Type;
21  
22  /**
23   * Contract for a provider of {@link ParamConverter} instances.
24   * <p>
25   * Providers implementing {@code ParamConverterProvider} contract must be either programmatically registered in a JAX-RS
26   * runtime or must be annotated with {@link jakarta.ws.rs.ext.Provider &#64;Provider} annotation to be automatically
27   * discovered by the JAX-RS runtime during a provider scanning phase.
28   * </p>
29   *
30   * @author Marek Potociar
31   * @since 2.0
32   */
33  public interface ParamConverterProvider {
34  
35      /**
36       * Obtain a {@link ParamConverter} that can provide from/to string conversion for an instance of a particular Java type.
37       *
38       * @param <T> the supported Java type convertible to/from a {@code String} format.
39       * @param rawType the raw type of the object to be converted.
40       * @param genericType the type of object to be converted. E.g. if an String value representing the injected request
41       * parameter is to be converted into a method parameter, this will be the formal type of the method parameter as
42       * returned by {@code Class.getGenericParameterTypes}.
43       * @param annotations an array of the annotations associated with the convertible parameter instance. E.g. if a string
44       * value is to be converted into a method parameter, this would be the annotations on that parameter as returned by
45       * {@link java.lang.reflect.Method#getParameterAnnotations}.
46       * @return the string converter, otherwise {@code null}.
47       */
48      public <T> ParamConverter<T> getConverter(Class<T> rawType, Type genericType, Annotation[] annotations);
49  }
