17  package jakarta.ws.rs.ext;
18  
19  import java.io.InputStream;
20  import java.lang.annotation.Annotation;
21  import java.lang.reflect.Type;
22  
23  import jakarta.ws.rs.core.MediaType;
24  import jakarta.ws.rs.core.MultivaluedMap;
25  
26  /**
27   * Contract for a provider that supports the conversion of a stream to a Java type.
28   *
29   * A {@code MessageBodyReader} implementation may be annotated with {@link jakarta.ws.rs.Consumes} to restrict the media
30   * types for which it will be considered suitable. The {@code MessageBodyReader} pipeline is executed if the matching
31   * resource method declares an entity parameter or uses at least one {@link jakarta.ws.rs.FormParam}.
32   * <p>
33   * Providers implementing {@code MessageBodyReader} contract must be either programmatically registered in a JAX-RS
34   * runtime or must be annotated with {@link jakarta.ws.rs.ext.Provider &#64;Provider} annotation to be automatically
35   * discovered by the JAX-RS runtime during a provider scanning phase.
36   * </p>
37   *
38   * @param <T> Java type supported by the provider
39   * @author Paul Sandoz
40   * @author Marc Hadley
41   * @see Provider
42   * @see jakarta.ws.rs.Consumes
43   * @since 1.0
44   */
45  public interface MessageBodyReader<T> {
46  
47      /**
48       * Ascertain if the MessageBodyReader can produce an instance of a particular type. The {@code type} parameter gives the
49       * class of the instance that should be produced, the {@code genericType} parameter gives the
50       * {@link java.lang.reflect.Type java.lang.reflect.Type} of the instance that should be produced. E.g. if the instance
51       * to be produced is {@code List<String>}, the {@code type} parameter will be {@code java.util.List} and the
52       * {@code genericType} parameter will be {@link java.lang.reflect.ParameterizedType
53       * java.lang.reflect.ParameterizedType}.
54       *
55       * @param type the class of instance to be produced.
56       * @param genericType the type of instance to be produced. E.g. if the message body is to be converted into a method
57       * parameter, this will be the formal type of the method parameter as returned by
58       * {@code Method.getGenericParameterTypes}.
59       * @param annotations an array of the annotations on the declaration of the artifact that will be initialized with the
60       * produced instance. E.g. if the message body is to be converted into a method parameter, this will be the annotations
61       * on that parameter returned by {@code Method.getParameterAnnotations}.
62       * @param mediaType the media type of the HTTP entity, if one is not specified in the request then
63       * {@code application/octet-stream} is used.
64       * @return {@code true} if the type is supported, otherwise {@code false}.
65       */
66      public boolean isReadable(Class<?> type, Type genericType,
67              Annotation[] annotations, MediaType mediaType);
68  
69      /**
70       * Read a type from the {@link InputStream}.
71       * <p>
72       * In case the entity input stream is empty, the reader is expected to either return a Java representation of a
73       * zero-length entity or throw a {@link jakarta.ws.rs.core.NoContentException} in case no zero-length entity
74       * representation is defined for the supported Java type. A {@code NoContentException}, if thrown by a message body
75       * reader while reading a server request entity, is automatically translated by JAX-RS server runtime into a
76       * {@link jakarta.ws.rs.BadRequestException} wrapping the original {@code NoContentException} and rethrown for a standard
77       * processing by the registered {@link jakarta.ws.rs.ext.ExceptionMapper exception mappers}.
78       * </p>
79       *
80       * @param type the type that is to be read from the entity stream.
81       * @param genericType the type of instance to be produced. E.g. if the message body is to be converted into a method
82       * parameter, this will be the formal type of the method parameter as returned by
83       * {@code Method.getGenericParameterTypes}.
84       * @param annotations an array of the annotations on the declaration of the artifact that will be initialized with the
85       * produced instance. E.g. if the message body is to be converted into a method parameter, this will be the annotations
86       * on that parameter returned by {@code Method.getParameterAnnotations}.
87       * @param mediaType the media type of the HTTP entity.
88       * @param httpHeaders the read-only HTTP headers associated with HTTP entity.
89       * @param entityStream the {@link InputStream} of the HTTP entity. The caller is responsible for ensuring that the input
90       * stream ends when the entity has been consumed. The implementation should not close the input stream.
91       * @return the type that was read from the stream. In case the entity input stream is empty, the reader is expected to
92       * either return an instance representing a zero-length entity or throw a {@link jakarta.ws.rs.core.NoContentException} in
93       * case no zero-length entity representation is defined for the supported Java type.
94       * @throws java.io.IOException if an IO error arises. In case the entity input stream is empty and the reader is not
95       * able to produce a Java representation for a zero-length entity, {@code NoContentException} is expected to be thrown.
96       * @throws jakarta.ws.rs.WebApplicationException if a specific HTTP error response needs to be produced. Only effective if
97       * thrown prior to the response being committed.
98       */
99      public T readFrom(Class<T> type, Type genericType,
100             Annotation[] annotations, MediaType mediaType,
101             MultivaluedMap<String, String> httpHeaders,
102             InputStream entityStream) throws java.io.IOException, jakarta.ws.rs.WebApplicationException;
103 }
