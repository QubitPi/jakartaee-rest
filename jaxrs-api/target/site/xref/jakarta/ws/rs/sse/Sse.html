17  package jakarta.ws.rs.sse;
18  
19  /**
20   * Server-side entry point for creating {@link OutboundSseEvent} and {@link SseBroadcaster}.
21   * <p>
22   * Instance of this interface can be injected into a field or as a parameter of a method or a constructor. Also, the
23   * instance is thread safe, meaning that it can be shared and its method invoked from different threads without causing
24   * inconsistent internal state.
25   *
26   * @author Marek Potociar (marek.potociar at oracle.com)
27   * @since 2.1
28   */
29  public interface Sse {
30  
31      /**
32       * Get a new outbound event builder.
33       *
34       * @return SSE outbound event builder.
35       */
36      OutboundSseEvent.Builder newEventBuilder();
37  
38      /**
39       * Create new {@link OutboundSseEvent} from provided data.
40       * <p>
41       * The data can be string only and must not be null.
42       *
43       * @param data event data.
44       * @return created {@code OutboundSseEvent}.
45       * @throws IllegalArgumentException when data is {@code null}.
46       */
47      default OutboundSseEvent newEvent(final String data) {
48          if (data == null) {
49              throw new IllegalArgumentException("Parameter 'data' must not be null.");
50          }
51  
52          return newEventBuilder().data(String.class, data).build();
53      }
54  
55      /**
56       * Create new {@link OutboundSseEvent} from provided data and name.
57       *
58       * @param name event name. (see {@link OutboundSseEvent#getName()}).
59       * @param data event data.
60       * @return created {@code OutboundSseEvent}.
61       * @throws IllegalArgumentException when name or data is {@code null}.
62       */
63      default OutboundSseEvent newEvent(final String name, final String data) {
64          if (data == null) {
65              throw new IllegalArgumentException("Parameter 'data' must not be null.");
66          }
67  
68          if (name == null) {
69              throw new IllegalArgumentException("Parameter 'name' must not be null.");
70          }
71  
72          return newEventBuilder().data(String.class, data).name(name).build();
73      }
74  
75      /**
76       * Get a new Server-sent event broadcaster.
77       *
78       * @return new Server-sent event broadcaster instance.
79       */
80      SseBroadcaster newBroadcaster();
81  }
