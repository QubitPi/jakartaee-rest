17  package jakarta.ws.rs.sse;
18  
19  import java.util.concurrent.CompletionStage;
20  import java.util.function.BiConsumer;
21  import java.util.function.Consumer;
22  
23  /**
24   * Server-Sent events broadcasting facility.
25   * <p>
26   * Server broadcaster can be used to manage multiple {@link SseEventSink server sinks}. It enables sending events to all
27   * registered event outputs and provides facility to effectively handle exceptions and closures of individual registered
28   * event outputs.
29   * <p>
30   * Instance of this interface is thread safe, meaning that it can be shared and its method invoked from different
31   * threads without causing inconsistent internal state.
32   *
33   * @author Marek Potociar
34   * @since 2.1
35   */
36  public interface SseBroadcaster extends AutoCloseable {
37  
38      /**
39       * Register a listener, which will be called when an exception is thrown by a given {@link SseEventSink}
40       * when this SseBroadcaster tries to write to it or close it.
41       * <p>
42       * This operation is potentially slow, especially if large number of listeners get registered in the broadcaster. The
43       * {@code SseBroadcaster} implementation is optimized to efficiently handle small amounts of concurrent listener
44       * registrations and removals and large amounts of registered listener notifications.
45       *
46       * @param onError bi-consumer, taking two parameters: {@link SseEventSink}, which is the source of the error and the
47       * actual {@link Throwable} instance.
48       */
49      void onError(BiConsumer<SseEventSink, Throwable> onError);
50  
51      /**
52       * Register a listener, which will be called when this SseBroadcaster closes a given event {@link SseEventSink} or
53       * tries to write to a given {@link SseEventSink} that is already closed (either by client closing the connection
54       * or by calling {@link SseEventSink#close()} on the server side.
55       * <p>
56       * This operation is potentially slow, especially if large number of listeners get registered in the broadcaster. The
57       * {@code SseBroadcaster} implementation is optimized to efficiently handle small amounts of concurrent listener
58       * registrations and removals and large amounts of registered listener notifications.
59       *
60       * @param onClose consumer taking single parameter, a {@link SseEventSink}, which was closed.
61       */
62      void onClose(Consumer<SseEventSink> onClose);
63  
64      /**
65       * Register provided {@link SseEventSink} instance to this {@code SseBroadcaster}.
66       *
67       * @param sseEventSink to be registered.
68       */
69      void register(SseEventSink sseEventSink);
70  
71      /**
72       * Publish an SSE event to all registered {@link SseEventSink} instances.
73       *
74       * @param event SSE event to be published.
75       * @return completion stage that completes when the event has been broadcast to all registered event sinks.
76       */
77      CompletionStage<?> broadcast(final OutboundSseEvent event);
78  
79      /**
80       * Close the broadcaster and all registered {@link SseEventSink} instances. Any other resources associated with the
81       * {@link SseBroadcaster} should be released. This method is equivalent to calling {@code close(true)}.
82       * <p>
83       * Subsequent calls have no effect and are ignored. Once the {@link SseBroadcaster} is closed, invoking any other method
84       * on the broadcaster instance would result in an {@link IllegalStateException} being thrown.
85       */
86      @Override
87      void close();
88  
89      /**
90       * Close the broadcaster and release any resources associated with it. The closing of registered {@link SseEventSink} is
91       * controlled by the {@code cascading} parameter.
92       * <p>
93       * Subsequent calls have no effect and are ignored. Once the {@link SseBroadcaster} is closed, invoking any other method
94       * on the broadcaster instance would result in an {@link IllegalStateException} being thrown.
95       *
96       * @param cascading Boolean value that controls closing of registered {@link SseEventSink} instances.
97       */
98      void close(boolean cascading);
99  }
