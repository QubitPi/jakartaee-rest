17  package jakarta.ws.rs.sse;
18  
19  import java.net.URL;
20  import java.util.concurrent.ScheduledExecutorService;
21  import java.util.concurrent.TimeUnit;
22  import java.util.function.Consumer;
23  
24  import jakarta.ws.rs.client.WebTarget;
25  
26  /**
27   * Client for reading and processing {@link InboundSseEvent incoming Server-Sent Events}.
28   * <p>
29   * SSE event source instances of this class are thread safe. To build a new instance, you can use the
30   * {@link #target(jakarta.ws.rs.client.WebTarget) SseEventSource.target(endpoint)} factory method to get a new event
31   * source builder that can be further customised and eventually used to create a new SSE event source.
32   * <p>
33   * Once a {@link SseEventSource} is created, it can be used to {@link #open open a connection} to the associated
34   * {@link WebTarget web target}. After establishing the connection, the event source starts processing any incoming
35   * inbound events. Whenever a new event is received, an {@link Consumer#accept(Object)
36   * Consumer&lt;InboundSseEvent&gt;#accept(InboundSseEvent)} method is invoked on any registered event consumers.
37   * <h2>Reconnect support</h2>
38   * <p>
39   * The {@code SseEventSource} supports automated recuperation from a connection loss, including
40   * negotiation of delivery of any missed events based on the last received  SSE event {@code id} field value, provided
41   * this field is set by the server and the negotiation facility is supported by the server. In case of a connection loss,
42   * the last received SSE event {@code id} field value is sent in the
43   * <code>{@value jakarta.ws.rs.core.HttpHeaders#LAST_EVENT_ID_HEADER}</code> HTTP
44   * request header as part of a new connection request sent to the SSE endpoint. Upon a receipt of such reconnect request, the SSE
45   * endpoint that supports this negotiation facility is expected to replay all missed events. Note however, that this is a
46   * best-effort mechanism which does not provide any guaranty that all events would be delivered without a loss. You should
47   * therefore not rely on receiving every single event and design your client application code accordingly.
48   * <p>
49   * By default, when a connection to the SSE endpoint is lost, the event source will wait 500&nbsp;ms
50   * before attempting to reconnect to the SSE endpoint. The SSE endpoint can however control the client-side retry delay
51   * by including a special {@code retry} field value in any sent event. JAX-RS {@code SseEventSource} tracks any
52   * received SSE event {@code retry} field values set by the endpoint and adjusts the reconnect delay accordingly,
53   * using the last received {@code retry} field value as the reconnect delay.
54   * <p>
55   * In addition to handling the standard connection loss failures, JAX-RS {@code SseEventSource} behaves differently to various
56   * HTTP response status codes and headers:
57   * <ul>
58   * <li>200 - with <code>{@value jakarta.ws.rs.core.HttpHeaders#CONTENT_TYPE}</code> header of "text/event-stream": This is normal
59   * operation. <code>onEvent</code> is invoked for each event.  <code>onComplete</code> is invoked when there are no more
60   * events. <code>onError</code> is invoked only if an unrecoverable error occurs during processing.</li>
61   * <li>200 - with unsupported or missing <code>{@value jakarta.ws.rs.core.HttpHeaders#CONTENT_TYPE}</code> header: This is an
62   * error condition. <code>onError</code> is invoked.</li>
63   * <li>204 - This indicates that server has no events to send. Only <code>onComplete</code> is invoked.</li>
64   * <li>503 - with <code>{@value jakarta.ws.rs.core.HttpHeaders#RETRY_AFTER}</code> header set to a valid value: This indicates
65   * that the server is unavailable, but that the client should reconnect later. No consumers are invoked unless the client
66   * event source is closed, prior to reconnecting (resulting in <code>onComplete</code> invocation). After the specified
67   * delay, the client should automatically attempt to reconnect which will result in a new response.</li>
68   * <li>503 - with invalid or missing <code>{@value jakarta.ws.rs.core.HttpHeaders#RETRY_AFTER}</code> header: This is an error
69   * condition. <code>onError</code> is invoked.</li>
70   * <li>Any other status code: This is an error condition. <code>onError</code> is invoked.</li>
71   * </ul>
72   * <p>
73   * In the case of an error condition response, the <code>Throwable</code> passed to the <code>onError</code> consumer
74   * <i>should</i> be a WebApplicationException containing the invalid <code>Response</code> object.
75   * <p>
76   * Note that if, for any of the registered event consumers, an invocation of {@link Consumer#accept(Object)
77   * Consumer&lt;InboundSseEvent&gt;#accept(InboundSseEvent)} method throws an exception, this is not an error condition.
78   * Thus <code>onError</code> is not invoked and event processing is not stopped.
79   * Users are encouraged to handle exceptions on their own as part of the event processing logic.
80   *
81   * @author Marek Potociar
82   * @since 2.1
83   */
84  public interface SseEventSource extends AutoCloseable {
85  
86      /**
87       * JAX-RS {@link SseEventSource} builder class.
88       * <p>
89       * Event source builder provides methods that let you conveniently configure and subsequently build a new
90       * {@code SseEventSource} instance. You can obtain a new event source builder instance using a static
91       * {@link SseEventSource#target(jakarta.ws.rs.client.WebTarget) SseEventSource.target(endpoint)} factory method.
92       * <p>
93       * For example:
94       *
95       * <pre>
96       * SseEventSource es = SseEventSource.target(endpoint)
97       *         .reconnectingEvery(5, SECONDS)
98       *         .build();
99       * es.register(System.out::println);
100      * es.open();
101      * </pre>
102      */
103     abstract class Builder {
104 
105         /**
106          * Name of the property identifying the {@link SseEventSource.Builder} implementation to be returned from
107          * {@link SseEventSource.Builder#newBuilder()}.
108          */
109         public static final String JAXRS_DEFAULT_SSE_BUILDER_PROPERTY = "jakarta.ws.rs.sse.SseEventSource.Builder";
110 
111         /**
112          * Allows custom implementations to extend the SSE event source builder class.
113          */
114         protected Builder() {
115         }
116 
117         /**
118          * Create a new SSE event source instance using the default implementation class provided by the JAX-RS implementation
119          * provider.
120          *
121          * @return new SSE event source builder instance.
122          */
123         static Builder newBuilder() {
124             try {
125                 Object delegate = FactoryFinder.find(JAXRS_DEFAULT_SSE_BUILDER_PROPERTY, SseEventSource.Builder.class);
126                 if (!(delegate instanceof Builder)) {
127                     Class pClass = Builder.class;
128                     String classnameAsResource = pClass.getName().replace('.', '/') + ".class";
129                     ClassLoader loader = pClass.getClassLoader();
130                     if (loader == null) {
131                         loader = ClassLoader.getSystemClassLoader();
132                     }
133                     URL targetTypeURL = loader.getResource(classnameAsResource);
134                     throw new LinkageError("ClassCastException: attempting to cast"
135                             + delegate.getClass().getClassLoader().getResource(classnameAsResource)
136                             + " to " + targetTypeURL);
137                 }
138                 return (Builder) delegate;
139             } catch (Exception ex) {
140                 throw new RuntimeException(ex);
141             }
142         }
143 
144         /**
145          * Set the SSE streaming endpoint.
146          *
147          * @param endpoint SSE streaming endpoint. Must not be {@code null}.
148          * @return updated event source builder instance.
149          * @throws NullPointerException in case the supplied web target is {@code null}.
150          */
151         protected abstract Builder target(WebTarget endpoint);
152 
153         /**
154          * Set the initial reconnect delay to be used by the event source.
155          * <p>
156          * Note that this value may be later overridden by the SSE endpoint using either a {@code retry} SSE event field or
157          * <code>HTTP 503 + {@value jakarta.ws.rs.core.HttpHeaders#RETRY_AFTER}</code> mechanism as described in the
158          * {@link SseEventSource} javadoc.
159          *
160          * @param delay the default time to wait before attempting to recover from a connection loss.
161          * @param unit time unit of the reconnect delay parameter.
162          * @return updated event source builder instance.
163          */
164         public abstract Builder reconnectingEvery(long delay, TimeUnit unit);
165 
166         /**
167          * Build new SSE event source pointing at a SSE streaming {@link WebTarget web target}.
168          * <p>
169          * The returned event source is ready, but not {@link SseEventSource#open() connected} to the SSE endpoint. It is
170          * expected that you will manually invoke its {@link #open()} method once you are ready to start receiving SSE events.
171          * In case you want to build an event source instance that is already connected to the SSE endpoint, use the event
172          * source builder {@link #open()} method instead.
173          * <p>
174          * Once the event source is open, the incoming events are processed by the event source in an asynchronous task that
175          * runs in an internal single-threaded {@link ScheduledExecutorService scheduled executor service}.
176          *
177          * @return new event source instance, ready to be connected to the SSE endpoint.
178          * @see #open()
179          */
180         public abstract SseEventSource build();
181     }
182 
183     /**
184      * Register a {@link InboundSseEvent} consumer.
185      * <p>
186      * Given consumer is invoked once per each received event.
187      *
188      * @param onEvent event consumer.
189      * @throws IllegalArgumentException when the provided parameter is {@code null}.
190      */
191     void register(Consumer<InboundSseEvent> onEvent);
192 
193     /**
194      * Register {@link InboundSseEvent} and {@link Throwable} consumers.
195      * <p>
196      * Event consumer is invoked once per each received event, {@code Throwable} consumer is invoked invoked upon a
197      * unrecoverable error encountered by a {@link SseEventSource}.
198      *
199      * @param onEvent event consumer.
200      * @param onError error consumer.
201      * @throws IllegalArgumentException when the any of the provided parameters is {@code null}.
202      */
203     void register(Consumer<InboundSseEvent> onEvent,
204             Consumer<Throwable> onError);
205 
206     /**
207      * Register {@link InboundSseEvent} and {@link Throwable} consumers and onComplete callback.
208      * <p>
209      * Event consumer is invoked once per each received event, {@code Throwable} consumer is invoked invoked upon a
210      * unrecoverable error encountered by a {@link SseEventSource}, onComplete callback is invoked after a successful
211      * connection and when there are no further events to be received. Note that the onComplete callback will not be
212      * invoked if the onError callback has been invoked.
213      *
214      * @param onEvent event consumer.
215      * @param onError error consumer.
216      * @param onComplete onComplete handler.
217      * @throws IllegalArgumentException when the any of the provided parameters is {@code null}.
218      */
219     void register(Consumer<InboundSseEvent> onEvent,
220             Consumer<Throwable> onError,
221             Runnable onComplete);
222 
223     /**
224      * Create a new {@link SseEventSource.Builder event source builder} that provides convenient way how to configure and
225      * fine-tune various aspects of a newly prepared event source instance.
226      *
227      * @param endpoint SSE streaming endpoint. Must not be {@code null}.
228      * @return a builder of a new event source instance pointing at the specified SSE streaming endpoint.
229      * @throws NullPointerException in case the supplied web target is {@code null}.
230      */
231     static Builder target(final WebTarget endpoint) {
232         return Builder.newBuilder().target(endpoint);
233     }
234 
235     /**
236      * Open the connection to the supplied SSE underlying {@link WebTarget web target} and start processing incoming
237      * {@link InboundSseEvent events}.
238      *
239      * @throws IllegalStateException in case the event source has already been opened earlier.
240      */
241     void open();
242 
243     /**
244      * Check if this event source instance has already been {@link #open() opened}.
245      *
246      * @return {@code true} if this event source is open, {@code false} otherwise.
247      */
248     boolean isOpen();
249 
250     /**
251      * Close this event source.
252      * <p>
253      * The method will wait up to 5 seconds for the internal event processing tasks to complete.
254      */
255     @Override
256     default void close() {
257         close(5, TimeUnit.SECONDS);
258     }
259 
260     /**
261      * Close this event source and wait for the internal event processing task to complete for up to the specified amount of
262      * wait time.
263      * <p>
264      * The method blocks until the event processing task has completed execution after a shutdown request, or until the
265      * timeout occurs, or the current thread is interrupted, whichever happens first.
266      * <p>
267      * In case the waiting for the event processing task has been interrupted, this method restores the
268      * {@link Thread#interrupted() interrupt} flag on the thread before returning {@code false}.
269      *
270      * @param timeout the maximum time to wait.
271      * @param unit the time unit of the timeout argument.
272      * @return {@code true} if this executor terminated and {@code false} if the timeout elapsed before termination or the
273      * termination was interrupted.
274      */
275     boolean close(final long timeout, final TimeUnit unit);
276 }
