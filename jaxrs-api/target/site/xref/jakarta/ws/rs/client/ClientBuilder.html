34  /**
35   * Main entry point to the client API used to bootstrap {@link jakarta.ws.rs.client.Client} instances.
36   *
37   * @author Marek Potociar
38   * @since 2.0
39   */
40  public abstract class ClientBuilder implements Configurable<ClientBuilder> {
41  
42      /**
43       * Name of the property identifying the {@link ClientBuilder} implementation to be returned from
44       * {@link ClientBuilder#newBuilder()}.
45       */
46      public static final String JAXRS_DEFAULT_CLIENT_BUILDER_PROPERTY = "jakarta.ws.rs.client.ClientBuilder";
47  
48      /**
49       * Allows custom implementations to extend the {@code ClientBuilder} class.
50       */
51      protected ClientBuilder() {
52      }
53  
54      /**
55       * Create a new {@code ClientBuilder} instance using the default client builder implementation class provided by the
56       * JAX-RS implementation provider.
57       *
58       * @return new client builder instance.
59       */
60      public static ClientBuilder newBuilder() {
61          try {
62              Object delegate = FactoryFinder.find(JAXRS_DEFAULT_CLIENT_BUILDER_PROPERTY, ClientBuilder.class);
63              if (!(delegate instanceof ClientBuilder)) {
64                  final CreateErrorMessageAction action = new CreateErrorMessageAction(delegate);
65                  final String errorMessage;
66                  if (System.getSecurityManager() == null) {
67                      errorMessage = action.run();
68                  } else {
69                      errorMessage = AccessController.doPrivileged(action);
70                  }
71                  throw new LinkageError(errorMessage);
72              }
73              return (ClientBuilder) delegate;
74          } catch (Exception ex) {
75              throw new RuntimeException(ex);
76          }
77      }
78  
79      /**
80       * Create a new {@link Client} instance using the default client builder implementation class provided by the JAX-RS
81       * implementation provider.
82       *
83       * @return new client instance.
84       */
85      public static Client newClient() {
86          return newBuilder().build();
87      }
88  
89      /**
90       * Create a new custom-configured {@link Client} instance using the default client builder implementation class provided
91       * by the JAX-RS implementation provider.
92       *
93       * @param configuration data used to provide initial configuration for the new client instance.
94       * @return new configured client instance.
95       */
96      public static Client newClient(final Configuration configuration) {
97          return newBuilder().withConfig(configuration).build();
98      }
99  
100     /**
101      * Set the internal configuration state to an externally provided configuration state.
102      *
103      * @param config external configuration state to replace the configuration of this configurable instance.
104      * @return the updated client builder instance.
105      */
106     public abstract ClientBuilder withConfig(Configuration config);
107 
108     /**
109      * Set the SSL context that will be used when creating secured transport connections to server endpoints from
110      * {@link WebTarget web targets} created by the client instance that is using this SSL context. The SSL context is
111      * expected to have all the security infrastructure initialized, including the key and trust managers.
112      * <p>
113      * Setting a SSL context instance resets any {@link #keyStore(java.security.KeyStore, char[]) key store} or
114      * {@link #trustStore(java.security.KeyStore) trust store} values previously specified.
115      * </p>
116      *
117      * @param sslContext secure socket protocol implementation which acts as a factory for secure socket factories or
118      * {@link javax.net.ssl.SSLEngine SSL engines}. Must not be {@code null}.
119      * @return an updated client builder instance.
120      * @throws NullPointerException in case the {@code sslContext} parameter is {@code null}.
121      * @see #keyStore(java.security.KeyStore, char[])
122      * @see #keyStore(java.security.KeyStore, String)
123      * @see #trustStore
124      */
125     public abstract ClientBuilder sslContext(final SSLContext sslContext);
126 
127     /**
128      * Set the client-side key store. Key store contains client's private keys, and the certificates with their
129      * corresponding public keys.
130      * <p>
131      * Setting a key store instance resets any {@link #sslContext(javax.net.ssl.SSLContext) SSL context instance} value
132      * previously specified.
133      * </p>
134      * <p>
135      * Note that a custom key store is only required if you want to enable a custom setup of a 2-way SSL connections (client
136      * certificate authentication).
137      * </p>
138      *
139      * @param keyStore client-side key store. Must not be {@code null}.
140      * @param password client key password. Must not be {@code null}.
141      * @return an updated client builder instance.
142      * @throws NullPointerException in case any of the supplied parameters is {@code null}.
143      * @see #sslContext
144      * @see #keyStore(java.security.KeyStore, String)
145      * @see #trustStore
146      */
147     public abstract ClientBuilder keyStore(final KeyStore keyStore, final char[] password);
148 
149     /**
150      * Set the client-side key store. Key store contains client's private keys, and the certificates with their
151      * corresponding public keys.
152      * <p>
153      * Setting a key store instance resets any {@link #sslContext(javax.net.ssl.SSLContext) SSL context instance} value
154      * previously specified.
155      * </p>
156      * <p>
157      * Note that for improved security of working with password data and avoid storing passwords in Java string objects, the
158      * {@link #keyStore(java.security.KeyStore, char[])} version of the method can be utilized. Also note that a custom key
159      * store is only required if you want to enable a custom setup of a 2-way SSL connections (client certificate
160      * authentication).
161      * </p>
162      *
163      * @param keyStore client-side key store. Must not be {@code null}.
164      * @param password client key password. Must not be {@code null}.
165      * @return an updated client builder instance.
166      * @throws NullPointerException in case any of the supplied parameters is {@code null}.
167      * @see #sslContext
168      * @see #keyStore(java.security.KeyStore, char[])
169      * @see #trustStore
170      */
171     public ClientBuilder keyStore(final KeyStore keyStore, final String password) {
172         return keyStore(keyStore, password.toCharArray());
173     }
174 
175     /**
176      * Set the client-side trust store. Trust store is expected to contain certificates from other parties the client is you
177      * expect to communicate with, or from Certificate Authorities that are trusted to identify other parties.
178      * <p>
179      * Setting a trust store instance resets any {@link #sslContext(javax.net.ssl.SSLContext) SSL context instance} value
180      * previously specified.
181      * </p>
182      * <p>
183      * In case a custom trust store or custom SSL context is not specified, the trust management will be configured to use
184      * the default Java runtime settings.
185      * </p>
186      *
187      * @param trustStore client-side trust store. Must not be {@code null}.
188      * @return an updated client builder instance.
189      * @throws NullPointerException in case the supplied trust store parameter is {@code null}.
190      * @see #sslContext
191      * @see #keyStore(java.security.KeyStore, char[])
192      * @see #keyStore(java.security.KeyStore, String)
193      */
194     public abstract ClientBuilder trustStore(final KeyStore trustStore);
195 
196     /**
197      * Set the hostname verifier to be used by the client to verify the endpoint's hostname against it's identification
198      * information.
199      *
200      * @param verifier hostname verifier.
201      * @return an updated client builder instance.
202      */
203     public abstract ClientBuilder hostnameVerifier(final HostnameVerifier verifier);
204 
205     /**
206      * Set the client-side {@link ExecutorService}.
207      * <p>
208      * Provided executor service will be used for executing asynchronous tasks.
209      * <p>
210      * When running in a Jakarta EE container, implementations are required to use the container-managed executor service by
211      * default. In Java SE, the default is implementation-specific. In either case, calling this method will override the
212      * default.
213      *
214      * @param executorService executor service to be used for async invocations.
215      * @return an updated client builder instance.
216      * @see Invocation.Builder#async()
217      * @see Invocation.Builder#rx()
218      * @see RxInvokerProvider#getRxInvoker(SyncInvoker, ExecutorService)
219      * @since 2.1
220      */
221     public abstract ClientBuilder executorService(final ExecutorService executorService);
222 
223     /**
224      * Set the client-side {@link ScheduledExecutorService}.
225      * <p>
226      * Provided executor service will be used for executing scheduled asynchronous tasks.
227      * <p>
228      * When running in a Jakarta EE container, implementations are required to use the container-managed scheduled executor
229      * service by default. In Java SE the default is implementation-specific. In either case, calling this method will
230      * override the default.
231      *
232      * @param scheduledExecutorService executor service to be used for scheduled async invocations.
233      * @return an updated client builder instance.
234      * @see jakarta.ws.rs.sse.SseEventSource.Builder#reconnectingEvery(long, TimeUnit)
235      * @since 2.1
236      */
237     public abstract ClientBuilder scheduledExecutorService(final ScheduledExecutorService scheduledExecutorService);
238 
239     /**
240      * Set the connect timeout.
241      * <p>
242      * Value {@code 0} represents infinity. Negative values are not allowed.
243      *
244      * @param timeout the maximum time to wait.
245      * @param unit the time unit of the timeout argument.
246      * @return an updated client builder instance.
247      * @throws IllegalArgumentException when the value is negative.
248      * @since 2.1
249      */
250     public abstract ClientBuilder connectTimeout(long timeout, TimeUnit unit);
251 
252     /**
253      * Set the read timeout.
254      * <p>
255      * The value is the timeout to read a response. If the server doesn't respond within the defined timeframe,
256      * {@link ProcessingException} is thrown with {@link TimeoutException} as a cause.
257      * <p>
258      * Value {@code 0} represents infinity. Negative values are not allowed.
259      *
260      * @param timeout the maximum time to wait.
261      * @param unit the time unit of the timeout argument.
262      * @return an updated client builder instance.
263      * @throws IllegalArgumentException when the value is negative.
264      * @since 2.1
265      */
266     public abstract ClientBuilder readTimeout(long timeout, TimeUnit unit);
267 
268     /**
269      * Build a new client instance using all the configuration previously specified in this client builder.
270      *
271      * @return a new client instance.
272      */
273     public abstract Client build();

275     private static final class CreateErrorMessageAction implements PrivilegedAction<String> {
276         private final Object delegate;
277 
278         private CreateErrorMessageAction(final Object delegate) {
279             this.delegate = delegate;
280         }
281 
282         @Override
283         public String run() {
284             Class<?> pClass = ClientBuilder.class;
285             String classnameAsResource = pClass.getName().replace('.', '/') + ".class";
286             ClassLoader loader = pClass.getClassLoader();
287             if (loader == null) {
288                 loader = ClassLoader.getSystemClassLoader();
289             }
290             URL targetTypeURL = loader.getResource(classnameAsResource);
291             return "ClassCastException: attempting to cast"
292                     + delegate.getClass().getClassLoader().getResource(classnameAsResource)
293                     + " to " + targetTypeURL;
294         }
295     }
