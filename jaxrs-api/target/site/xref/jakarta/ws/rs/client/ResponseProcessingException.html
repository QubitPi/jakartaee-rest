17  package jakarta.ws.rs.client;
18  
19  import jakarta.ws.rs.ProcessingException;
20  import jakarta.ws.rs.core.Response;
21  
22  /**
23   * JAX-RS client-side runtime processing exception thrown to indicate that response processing has failed (e.g. in a
24   * filter chain or during message entity de-serialization). The exception contains the nested {@link Response} instance
25   * for which the runtime response processing failed.
26   *
27   * @author Marek Potociar (marek.potociar at oracle.com)
28   * @since 2.0
29   */
30  public class ResponseProcessingException extends ProcessingException {
31  
32      private static final long serialVersionUID = -4923161617935731839L;
33  
34      /**
35       * The response instance for which the processing failed.
36       */
37      private final Response response;
38  
39      /**
40       * Constructs a new JAX-RS runtime response processing exception for a specific {@link Response response} with the
41       * specified cause and a detail message of {@code (cause==null ? null : cause.toString())} (which typically contains the
42       * class and detail message of {@code cause}). This constructor is useful for runtime exceptions that are little more
43       * than wrappers for other throwables.
44       *
45       * @param response the response instance for which the processing failed.
46       * @param cause the cause (which is saved for later retrieval by the {@link #getCause()} method). (A {@code null} value
47       * is permitted, and indicates that the cause is nonexistent or unknown.)
48       */
49      public ResponseProcessingException(final Response response, final Throwable cause) {
50          super(cause);
51          this.response = response;
52      }
53  
54      /**
55       * <p>
56       * Constructs a new JAX-RS runtime response processing exception with the specified detail message and cause.
57       * </p>
58       * Note that the detail message associated with {@code cause} is <i>not</i> automatically incorporated in this runtime
59       * exception's detail message.
60       *
61       * @param response the response instance for which the processing failed.
62       * @param message the detail message (which is saved for later retrieval by the {@link #getMessage()} method).
63       * @param cause the cause (which is saved for later retrieval by the {@link #getCause()} method). (A {@code null} value
64       * is permitted, and indicates that the cause is nonexistent or unknown.)
65       */
66      public ResponseProcessingException(final Response response, final String message, final Throwable cause) {
67          super(message, cause);
68          this.response = response;
69      }
70  
71      /**
72       * Constructs a new JAX-RS runtime processing exception with the specified detail message. The cause is not initialized,
73       * and may subsequently be initialized by a call to {@link #initCause}.
74       *
75       * @param response the response instance for which the processing failed.
76       * @param message the detail message (which is saved for later retrieval by the {@link #getMessage()} method).
77       */
78      public ResponseProcessingException(final Response response, final String message) {
79          super(message);
80          this.response = response;
81      }
82  
83      /**
84       * Get the HTTP response for which the processing has failed.
85       *
86       * @return the HTTP response.
87       */
88      public Response getResponse() {
89          return response;
90      }
91  }
