17  package jakarta.ws.rs.client;
18  
19  import java.io.IOException;
20  
21  /**
22   * An extension interface implemented by client request filters.
23   *
24   * Filters implementing this interface MUST be annotated with {@link jakarta.ws.rs.ext.Provider &#64;Provider} to be
25   * discovered by the JAX-RS runtime. This type of filters is supported only as part of the Client API.
26   *
27   * @author Marek Potociar
28   * @author Santiago Pericas-Geertsen
29   * @see jakarta.ws.rs.client.ClientResponseFilter
30   * @since 2.0
31   */
32  public interface ClientRequestFilter {
33  
34      /**
35       * Filter method called before a request has been dispatched to a client transport layer.
36       *
37       * Filters in the filter chain are ordered according to their {@code jakarta.annotation.Priority} class-level annotation
38       * value.
39       *
40       * @param requestContext request context.
41       * @throws IOException if an I/O exception occurs.
42       */
43      public void filter(ClientRequestContext requestContext) throws IOException;
44  }
