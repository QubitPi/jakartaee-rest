17  package jakarta.ws.rs.client;
18  
19  import java.util.concurrent.ExecutorService;
20  
21  /**
22   * {@link RxInvoker} provider.
23   * <p>
24   * {@code RxInvokerProvider} must be registered in the client runtime using {@link Client#register(Class)}. It provides
25   * a way to plug-in support for other reactive implementations, see {@link Invocation.Builder#rx(Class)}.
26   *
27   * @param <T> {@code RxInvoker} subclass type.
28   * @author Pavel Bucek
29   * @author Santiago Pericas-Geertsen
30   * @since 2.1
31   */
32  public interface RxInvokerProvider<T extends RxInvoker> {
33  
34      /**
35       * Determine if this is a provider for the given {@link RxInvoker} subclass.
36       *
37       * @param clazz {@code RxInvoker} subclass.
38       * @return {@code true} when this provider provides given {@code RxInvoker} subclass, {@code false} otherwise.
39       */
40      public boolean isProviderFor(Class<?> clazz);
41  
42      /**
43       * Get {@link RxInvoker} implementation instance.
44       * <p>
45       * The returned instance has to be thread safe.
46       *
47       * @param syncInvoker {@code SyncInvoker} used to execute current request.
48       * @param executorService executor service, which should be used for executing reactive callbacks invocations. It can be
49       * {@code null}; in that case it's up to the implementation to choose the best {@code ExecutorService} in given
50       * environment.
51       * @return instance of the {@code RxInvoker} subclass.
52       * @see ClientBuilder#executorService(ExecutorService)
53       */
54      public T getRxInvoker(SyncInvoker syncInvoker, ExecutorService executorService);
55  
56  }
