17  package jakarta.ws.rs.client;
18  
19  import java.net.URI;
20  
21  import javax.net.ssl.HostnameVerifier;
22  import javax.net.ssl.SSLContext;
23  
24  import jakarta.ws.rs.core.Configurable;
25  import jakarta.ws.rs.core.Link;
26  import jakarta.ws.rs.core.UriBuilder;
27  
28  /**
29   * <p>
30   * Client is the main entry point to the fluent API used to build and execute client requests in order to consume
31   * responses returned.
32   * </p>
33   * Clients are heavy-weight objects that manage the client-side communication infrastructure. Initialization as well as
34   * disposal of a {@code Client} instance may be a rather expensive operation. It is therefore advised to construct only
35   * a small number of {@code Client} instances in the application. Client instances must be {@link #close() properly
36   * closed} before being disposed to avoid leaking resources.
37   *
38   * @author Marek Potociar
39   * @see jakarta.ws.rs.core.Configurable
40   * @since 2.0
41   */
42  public interface Client extends Configurable<Client>, AutoCloseable {
43  
44      /**
45       * <p>
46       * Close client instance and all it's associated resources. Subsequent calls have no effect and are ignored. Once the
47       * client is closed, invoking any other method on the client instance would result in an {@link IllegalStateException}
48       * being thrown.
49       * </p>
50       * Calling this method effectively invalidates all {@link WebTarget resource targets} produced by the client instance.
51       * Invoking any method on such targets once the client is closed would result in an {@link IllegalStateException} being
52       * thrown.
53       */
54      @Override
55      public void close();
56  
57      /**
58       * Build a new web resource target.
59       *
60       * @param uri web resource URI. May contain template parameters. Must not be {@code null}.
61       * @return web resource target bound to the provided URI.
62       * @throws IllegalArgumentException in case the supplied string is not a valid URI template.
63       * @throws NullPointerException in case the supplied argument is {@code null}.
64       */
65      public WebTarget target(String uri);
66  
67      /**
68       * Build a new web resource target.
69       *
70       * @param uri web resource URI. Must not be {@code null}.
71       * @return web resource target bound to the provided URI.
72       * @throws NullPointerException in case the supplied argument is {@code null}.
73       */
74      public WebTarget target(URI uri);
75  
76      /**
77       * Build a new web resource target.
78       *
79       * @param uriBuilder web resource URI represented as URI builder. Must not be {@code null}.
80       * @return web resource target bound to the provided URI.
81       * @throws NullPointerException in case the supplied argument is {@code null}.
82       */
83      public WebTarget target(UriBuilder uriBuilder);
84  
85      /**
86       * Build a new web resource target.
87       *
88       * @param link link to a web resource. Must not be {@code null}.
89       * @return web resource target bound to the linked web resource.
90       * @throws NullPointerException in case the supplied argument is {@code null}.
91       */
92      public WebTarget target(Link link);
93  
94      /**
95       * <p>
96       * Build an invocation builder from a link. It uses the URI and the type of the link to initialize the invocation
97       * builder. The type is used as the initial value for the HTTP Accept header, if present.
98       * </p>
99       *
100      * @param link link to build invocation from. Must not be {@code null}.
101      * @return newly created invocation builder.
102      * @throws NullPointerException in case link is {@code null}.
103      */
104     public Invocation.Builder invocation(Link link);
105 
106     /**
107      * Get the SSL context configured to be used with the current client run-time.
108      *
109      * @return SSL context configured to be used with the current client run-time.
110      */
111     public SSLContext getSslContext();
112 
113     /**
114      * Get the hostname verifier configured in the client or {@code null} in case no hostname verifier has been configured.
115      *
116      * @return client hostname verifier or {@code null} if not set.
117      */
118     public HostnameVerifier getHostnameVerifier();
119 }
