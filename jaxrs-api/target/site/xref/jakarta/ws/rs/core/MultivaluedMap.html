17  package jakarta.ws.rs.core;
18  
19  import java.util.List;
20  import java.util.Map;
21  
22  /**
23   * A map of key-values pairs. Each key can have zero or more values.
24   *
25   * @param <K> the type of keys maintained by this map
26   * @param <V> the type of mapped values
27   * @author Paul Sandoz
28   * @author Marc Hadley
29   * @author Marek Potociar
30   * @since 1.0
31   */
32  public interface MultivaluedMap<K, V> extends Map<K, List<V>> {
33  
34      /**
35       * Set the key's value to be a one item list consisting of the supplied value. Any existing values will be replaced.
36       *
37       * @param key the key
38       * @param value the single value of the key
39       */
40      void putSingle(K key, V value);
41  
42      /**
43       * Add a value to the current list of values for the supplied key.
44       *
45       * @param key the key
46       * @param value the value to be added.
47       */
48      void add(K key, V value);
49  
50      /**
51       * A shortcut to get the first value of the supplied key.
52       *
53       * @param key the key
54       * @return the first value for the specified key or null if the key is not in the map.
55       */
56      V getFirst(K key);
57  
58      /**
59       * Add multiple values to the current list of values for the supplied key. If the supplied array of new values is empty,
60       * method returns immediately. Method throws a {@code NullPointerException} if the supplied array of values is
61       * {@code null}.
62       *
63       * @param key the key.
64       * @param newValues the values to be added.
65       * @throws NullPointerException if the supplied array of new values is {@code null}.
66       * @since 2.0
67       */
68      void addAll(K key, V... newValues);
69  
70      /**
71       * Add all the values from the supplied value list to the current list of values for the supplied key. If the supplied
72       * value list is empty, method returns immediately. Method throws a {@code NullPointerException} if the supplied array
73       * of values is {@code null}.
74       *
75       * @param key the key.
76       * @param valueList the list of values to be added.
77       * @throws NullPointerException if the supplied value list is {@code null}.
78       * @since 2.0
79       */
80      void addAll(K key, List<V> valueList);
81  
82      /**
83       * Add a value to the first position in the current list of values for the supplied key.
84       *
85       * @param key the key
86       * @param value the value to be added.
87       * @since 2.0
88       */
89      void addFirst(K key, V value);
90  
91      /**
92       * Compare the specified map with this map for equality modulo the order of values for each key. Specifically, the
93       * values associated with each key are compared as if they were ordered lists.
94       *
95       * @param otherMap map to be compared to this one.
96       * @return true if the maps are equal modulo value ordering.
97       * @since 2.0
98       */
99      boolean equalsIgnoreValueOrder(MultivaluedMap<K, V> otherMap);
100 
101 }
