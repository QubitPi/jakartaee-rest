17  package jakarta.ws.rs.core;
18  
19  import java.util.Collections;
20  import java.util.Map;
21  import java.util.Set;
22  
23  /**
24   * Defines the components of a JAX-RS application and supplies additional meta-data. A JAX-RS application or
25   * implementation supplies a concrete subclass of this abstract class.
26   * <p>
27   * The implementation-created instance of an Application subclass may be injected into resource classes and providers
28   * using {@link jakarta.ws.rs.core.Context}.
29   * </p>
30   * <p>
31   * In case any of the {@code Application} subclass methods or it's constructor throws a {@link RuntimeException}, the
32   * deployment of the application SHOULD be aborted with a failure.
33   * </p>
34   *
35   * @author Paul Sandoz
36   * @author Marc Hadley
37   * @author Marek Potociar
38   * @since 1.0
39   */
40  public class Application {
41  
42      /**
43       * Get a set of root resource, provider and {@link Feature feature} classes.
44       *
45       * The default life-cycle for resource class instances is per-request. The default life-cycle for providers (registered
46       * directly or via a feature) is singleton.
47       * <p>
48       * Implementations should warn about and ignore classes that do not conform to the requirements of root resource or
49       * provider/feature classes. Implementations should warn about and ignore classes for which {@link #getSingletons()}
50       * returns an instance. Implementations MUST NOT modify the returned set.
51       * </p>
52       * <p>
53       * The default implementation returns an empty set.
54       * </p>
55       *
56       * @return a set of root resource and provider classes. Returning {@code null} is equivalent to returning an empty set.
57       */
58      public Set<Class<?>> getClasses() {
59          return Collections.emptySet();
60      }
61  
62      /**
63       * Get a set of root resource, provider and {@link Feature feature} instances.
64       *
65       * Fields and properties of returned instances are injected with their declared dependencies (see {@link Context}) by
66       * the runtime prior to use.
67       * <p>
68       * Implementations should warn about and ignore classes that do not conform to the requirements of root resource or
69       * provider classes. Implementations should flag an error if the returned set includes more than one instance of the
70       * same class. Implementations MUST NOT modify the returned set.
71       * </p>
72       * <p>
73       * The default implementation returns an empty set.
74       * </p>
75       *
76       * @return a set of root resource and provider instances. Returning {@code null} is equivalent to returning an empty
77       * set.
78       * @deprecated Automatic discovery of resources and providers or the {@code getClasses} method is preferred over
79        * {@code getSingletons}.
80       */
81      @Deprecated
82      public Set<Object> getSingletons() {
83          return Collections.emptySet();
84      }
85  
86      /**
87       * Get a map of custom application-wide properties.
88       * <p>
89       * The returned properties are reflected in the application {@link Configuration configuration} passed to the
90       * server-side features or injected into server-side JAX-RS components.
91       * </p>
92       * <p>
93       * The set of returned properties may be further extended or customized at deployment time using container-specific
94       * features and deployment descriptors. For example, in a Servlet-based deployment scenario, web application's
95       * {@code <context-param>} and Servlet {@code <init-param>} values may be used to extend or override values of the
96       * properties programmatically returned by this method.
97       * </p>
98       * <p>
99       * The default implementation returns an empty set.
100      * </p>
101      *
102      * @return a map of custom application-wide properties. Returning {@code null} is equivalent to returning an empty set.
103      * @since 2.0
104      */
105     public Map<String, Object> getProperties() {
106         return Collections.emptyMap();
107     }
108 }
