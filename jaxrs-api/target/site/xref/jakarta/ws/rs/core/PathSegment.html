17  package jakarta.ws.rs.core;
18  
19  /**
20   * Represents a URI path segment and any associated matrix parameters. When an instance of this type is injected with
21   * {@link jakarta.ws.rs.PathParam}, the value of the annotation identifies which path segment is selected and the presence
22   * of an {@link jakarta.ws.rs.Encoded} annotation will result in an instance that supplies the path and matrix parameter
23   * values in URI encoded form.
24   *
25   * @author Paul Sandoz
26   * @author Marc Hadley
27   * @see UriInfo#getPathSegments
28   * @see jakarta.ws.rs.PathParam
29   * @since 1.0
30   */
31  public interface PathSegment {
32  
33      /**
34       * Get the path segment.
35       *
36       * @return the path segment
37       */
38      String getPath();
39  
40      /**
41       * Get a map of the matrix parameters associated with the path segment. The map keys are the names of the matrix
42       * parameters with any percent-escaped octets decoded.
43       *
44       * @return the map of matrix parameters
45       * @see <a href="http://www.w3.org/DesignIssues/MatrixURIs.html">Matrix URIs</a>
46       */
47      MultivaluedMap<String, String> getMatrixParameters();
48  }
