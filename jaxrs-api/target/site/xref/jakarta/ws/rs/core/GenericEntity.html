21  package jakarta.ws.rs.core;
22  
23  import java.lang.reflect.GenericArrayType;
24  import java.lang.reflect.ParameterizedType;
25  import java.lang.reflect.Type;
26  import java.util.Objects;
27  
28  /**
29   * Represents a message entity of a generic type {@code T}.
30   * <p>
31   * Normally type erasure removes generic type information such that a {@link Response} instance that contains, e.g., an
32   * entity of type {@code List<String>} appears to contain a raw {@code List<?>} at runtime. When the generic type is
33   * required to select a suitable {@link jakarta.ws.rs.ext.MessageBodyWriter}, this class may be used to wrap the entity
34   * and capture its generic type.
35   * </p>
36   * <p>
37   * There are two ways to create an instance:
38   * </p>
39   * <ol>
40   * <li>Create a (typically anonymous) subclass of this class which enables retrieval of the type information at runtime
41   * despite type erasure. For example, the following code shows how to create a {@link Response} containing an entity of
42   * type {@code List<String>} whose generic type will be available at runtime for selection of a suitable
43   * {@link jakarta.ws.rs.ext.MessageBodyWriter}:
44   *
45   * <pre>
46   * List&lt;String&gt; list = new ArrayList&lt;String&gt;();
47   * GenericEntity&lt;List&lt;String&gt;&gt; entity = new GenericEntity&lt;List&lt;String&gt;&gt;(list) {
48   * };
49   * Response response = Response.ok(entity).build();
50   * </pre>
51   *
52   * <p>
53   * where {@code list} is the instance of {@code List<String>} that will form the response body and entity is an instance
54   * of an anonymous subclass of {@code GenericEntity}.
55   * </p>
56   * </li>
57   * <li>Create an instance directly by supplying the generic type information with the entity. For example the following
58   * code shows how to create a response containing the result of a method invoked via reflection:
59   *
60   * <pre>
61   * Method method = ...;
62   * GenericEntity&lt;Object&gt; entity = new GenericEntity&lt;Object&gt;(
63   *    method.invoke(...), method.getGenericReturnType());
64   * Response response = Response.ok(entity).build();
65   * </pre>
66   * <p>
67   * The above obtains the generic type from the return type of the method, the raw type is the class of entity.
68   * </p>
69   * </li>
70   * </ol>
71   *
72   * @param <T> response entity instance type
73   * @author Paul Sandoz
74   * @author Marc Hadley
75   * @see GenericType
76   * @since 1.0
77   */
78  public class GenericEntity<T> {
79  
80      private final Class<?> rawType;
81      private final Type type;
82      private final T entity;
83  
84      /**
85       * Constructs a new generic entity. Derives represented class from type parameter. Note that this constructor is
86       * protected, users should create a (usually anonymous) subclass as shown above.
87       *
88       * @param entity the entity instance, must not be {@code null}.
89       * @throws IllegalArgumentException if entity is {@code null}.
90       */
91      protected GenericEntity(final T entity) {
92          if (entity == null) {
93              throw new IllegalArgumentException("The entity must not be null");
94          }
95          this.entity = entity;
96          this.type = GenericType.getTypeArgument(getClass(), GenericEntity.class);
97          this.rawType = entity.getClass();
98      }
99  
100     /**
101      * Create a new instance of GenericEntity, supplying the generic type information. The entity must be assignable to a
102      * variable of the supplied generic type, e.g. if {@code entity} is an instance of {@code ArrayList<String>} then
103      * {@code genericType} could be the same or a superclass of {@code ArrayList} with the same generic type like
104      * {@code List<String>}.
105      *
106      * @param entity the entity instance, must not be {@code null}.
107      * @param genericType the generic type, must not be {@code null}.
108      * @throws IllegalArgumentException if the entity is not assignable to a variable of the supplied generic type or if
109      * entity or genericType is null.
110      */
111     public GenericEntity(final T entity, final Type genericType) {
112         if (entity == null || genericType == null) {
113             throw new IllegalArgumentException("Arguments must not be null.");
114         }
115         this.entity = entity;
116         this.rawType = entity.getClass();
117         checkTypeCompatibility(this.rawType, genericType);
118         this.type = genericType;
119     }
120 
121     private void checkTypeCompatibility(final Class<?> c, final Type t) {
122         if (t instanceof Class) {
123             Class<?> ct = (Class<?>) t;
124             if (ct.isAssignableFrom(c)) {
125                 return;
126             }
127         } else if (t instanceof ParameterizedType) {
128             ParameterizedType pt = (ParameterizedType) t;
129             Type rt = pt.getRawType();
130             checkTypeCompatibility(c, rt);
131             return;
132         } else if (c.isArray() && (t instanceof GenericArrayType)) {
133             GenericArrayType at = (GenericArrayType) t;
134             Type rt = at.getGenericComponentType();
135             checkTypeCompatibility(c.getComponentType(), rt);
136             return;
137         }
138         throw new IllegalArgumentException("The type is incompatible with the class of the entity.");
139     }
140 
141     /**
142      * Gets the raw type of the enclosed entity. Note that this is the raw type of the instance, not the raw type of the
143      * type parameter. I.e. in the example in the introduction, the raw type is {@code ArrayList} not {@code List}.
144      *
145      * @return the raw type.
146      */
147     public final Class<?> getRawType() {
148         return rawType;
149     }
150 
151     /**
152      * Gets underlying {@code Type} instance. Note that this is derived from the type parameter, not the enclosed instance.
153      * I.e. in the example in the introduction, the type is {@code List<String>} not {@code ArrayList<String>}.
154      *
155      * @return the type
156      */
157     public final Type getType() {
158         return type;
159     }
160 
161     /**
162      * Get the enclosed entity.
163      *
164      * @return the enclosed entity.
165      */
166     public final T getEntity() {
167         return entity;
168     }
169 
170     @Override
171     public boolean equals(final Object obj) {
172         boolean result = this == obj;
173         if (!result && obj instanceof GenericEntity) {
174             // Compare inner type for equality
175             GenericEntity<?> that = (GenericEntity<?>) obj;
176             return this.type.equals(that.type) && this.entity.equals(that.entity);
177         }
178         return result;
179     }
180 
181     @Override
182     public int hashCode() {
183         return Objects.hash(entity, type);
184     }
185 
186     @Override
187     public String toString() {
188         return "GenericEntity{" + entity.toString() + ", " + type.toString() + "}";
189     }
190 }
