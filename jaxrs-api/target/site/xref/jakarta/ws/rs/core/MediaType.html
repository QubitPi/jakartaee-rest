169      * Creates a new instance of {@code MediaType} by parsing the supplied string.

173      * @throws IllegalArgumentException if the supplied string cannot be parsed or is {@code null}.

179     private static TreeMap<String, String> createParametersMap(final Map<String, String> initialValues) {
180         final TreeMap<String, String> map = new TreeMap<>(String.CASE_INSENSITIVE_ORDER);
181         if (initialValues != null) {
182             for (Map.Entry<String, String> e : initialValues.entrySet()) {
183                 map.put(e.getKey().toLowerCase(), e.getValue());
184             }
185         }
186         return map;
187     }

231     private MediaType(final String type, final String subtype, final String charset, final Map<String, String> parameterMap) {
232 
233         this.type = type == null ? MEDIA_TYPE_WILDCARD : type;
234         this.subtype = subtype == null ? MEDIA_TYPE_WILDCARD : subtype;
235 
236         Map<String, String> map = parameterMap;
237         if (map == null) {
238             map = new TreeMap<>(String.CASE_INSENSITIVE_ORDER);
239         }
240 
241         if (charset != null && !charset.isEmpty()) {
242             map.put(CHARSET_PARAMETER, charset);
243         }
244         this.parameters = Collections.unmodifiableMap(map);
245         this.hash = Objects.hash(this.type.toLowerCase(), this.subtype.toLowerCase(), this.parameters);
246     }

294      * Create a new {@code MediaType} instance with the same type, subtype and parameters copied from the original instance
295      * and the supplied {@value #CHARSET_PARAMETER} parameter.
296      *
297      * @param charset the {@value #CHARSET_PARAMETER} parameter value. If {@code null} or empty the
298      * {@value #CHARSET_PARAMETER} parameter will not be set or updated.
299      * @return copy of the current {@code MediaType} instance with the {@value #CHARSET_PARAMETER} parameter set to the

303     public MediaType withCharset(final String charset) {
304         return new MediaType(this.type, this.subtype, charset, createParametersMap(this.parameters));
305     }

307     /**
308      * Check if this media type is compatible with another media type.
309      * Two media types are considered to be compatible if and only if their types are equal,
310      * or one of them has a wildcard type, and their subtypes are equal or one of them has a wildcard subtype.
311      *
312      * Media type parameters are ignored. The function is commutative.
313      *
314      * @param other the media type to compare with.
315      * @return true if the types are compatible, false otherwise.
316      */
317     public boolean isCompatible(final MediaType other) {
318         if (other == null) {
319             return false;
320         }
321 
322         return
323             (type.equalsIgnoreCase(other.type) || this.isWildcardType() || other.isWildcardType())
324             &&
325             (subtype.equalsIgnoreCase(other.subtype) || this.isWildcardSubtype()
326              || other.isWildcardSubtype());
327     }

329     /**
330      * <p>
331      * Compares {@code obj} to this media type to see if they are the same by comparing type, subtype and parameters. Note
332      * that the case-sensitivity of parameter values is dependent on the semantics of the parameter name, see
333      * <a href="http://www.w3.org/Protocols/rfc2616/rfc2616-sec3.html#sec3.7">HTTP/1.1</a>. This method assumes that values
334      * are case-sensitive.
335      * </p>
336      * Note that the {@code equals(...)} implementation does not perform a class equality check
337      * ({@code this.getClass() == obj.getClass()}). Therefore any class that extends from {@code MediaType} class and needs
338      * to override one of the {@code equals(...)} and {@link #hashCode()} methods must always override both methods to
339      * ensure the contract between {@link Object#equals(java.lang.Object)} and {@link Object#hashCode()} does not break.
340      *
341      * @param obj the object to compare to.
342      * @return true if the two media types are the same, false otherwise.
343      */
344     @SuppressWarnings("UnnecessaryJavaDocLink")
345     @Override
346     public boolean equals(final Object obj) {
347         if (obj == this) {
348             return true;
349         }
350         if (!(obj instanceof MediaType)) {
351             return false;
352         }
353 
354         MediaType other = (MediaType) obj;
355         return (this.type.equalsIgnoreCase(other.type)
356                 && this.subtype.equalsIgnoreCase(other.subtype)
357                 && this.parameters.equals(other.parameters));
358     }
