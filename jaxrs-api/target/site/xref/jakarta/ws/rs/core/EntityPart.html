10  import java.io.IOException;
11  import java.io.InputStream;
12  import java.util.Optional;
13  
14  import jakarta.ws.rs.WebApplicationException;
15  import jakarta.ws.rs.ext.RuntimeDelegate;

17  /**
18   * A {@code EntityPart} is one part of a multipart entity. As defined in
19   * <a href="https://tools.ietf.org/html/rfc7578">RFC 7578</a>, a multipart
20   * request or response must have a content type of "multipart/form-data" with a
21   * {@code boundary} parameter indicating where one part ends the next may begin.
22   * <p>
23   * Multipart entities may be received in a resource method as a collection of
24   * parts (e.g. {@code List<EntityPart>}) or as a form parameter (ex:
25   * {@code @FormParam("part1Name") EntityPart part1}).
26   * </p>
27   * <p>
28   * Likewise, a client may receive a multipart response by reading the returned
29   * entity as a collection of EntityParts (ex: {@code response.readEntity(new
30   * GenericType<List<EntityPart>>() {})}).
31   * </p>
32   * <p>
33   * In order to send a multipart entity either as a client request or a response
34   * from a resource method, you may create the Lists using
35   * {@code EntityPart.Builder}. For example:
36   * </p>
37   *
38   * <pre>
39   * Client c = ClientBuilder.newClient();
40   *WebTarget target = c.target(someURL);
41   *List&lt;EntityPart&gt; parts = Arrays.asList(
42   *     EntityPart.withName("name1").fileName("file1.doc").content(stream1).build(),
43   *     EntityPart.withName("name2").fileName("file2.doc").content(stream2).build(),
44   *     EntityPart.withName("name3").fileName("file3.xml").content(myObject, MyClass.class).mediaType("application/xml").build());
45   *GenericEntity&lt;List&lt;EntityPart&gt;&gt; genericEntity = new GenericEntity&lt;&gt;(parts){};
46   *Entity entity = Entity.entity(genericEntity, MediaType.MULTIPART_FORM_DATA);
47   *Response r = target.request().post(entity);
48   * </pre>
49   *
50   * Note that when building a EntityPart, the name and content are required.
51   * Other properties such as headers, file name, and media type are optional.
52   *
53   * It is the responsibility of the implementation code to close the content
54   * input streams when sending the multipart content. Closing the stream before
55   * the implementation has sent it could result in unexpected exceptions. It is
56   * the responsibility of the calling code to close the stream when receiving the
57   * multipart content.
58   *
59   * @since 3.1
60   */
61  public interface EntityPart {
62  
63      /**
64       * Creates a new {@code EntityPart.Builder} instance.
65       *
66       * @param partName name of the part to create within the multipart entity
67       * @return {@link Builder} for building new {@link EntityPart} instances
68       */
69      static Builder withName(String partName) {
70          return RuntimeDelegate.getInstance().createEntityPartBuilder(partName);
71      }
72  
73      /**
74       * Creates a new {@code EntityPart.Builder} instance that sets the part
75       * {@code name} and {@code fileName} to the passed in {@code partAndFileName}
76       * value.
77       * <p>
78       * Logically, this is the same as {@code EntityPart.withName(x).fileName(x)}.
79       * </p>
80       *
81       * @param partAndFileName name and filename of the part to create within the
82       *                        multipart entity
83       * @return {@link Builder} for building new {@link EntityPart} instances
84       */
85      static Builder withFileName(String partAndFileName) {
86          return RuntimeDelegate.getInstance().createEntityPartBuilder(partAndFileName).fileName(partAndFileName);
87      }
88  
89      /**
90       * Returns the name of this part within the multipart entity. This will be the
91       * "name" attribute of the {@code Content-Disposition} header for this part.
92       *
93       * @return the part name
94       */
95      String getName();
96  
97      /**
98       * Returns the filename of this part. This will be the "filename" attribute of
99       * the {@code Content-Disposition} header for this part. A filename is not
100      * required in a part, so if a filename is not present it will return
101      * {@code Optional.empty()}.
102      *
103      * @return an {@code Optional<String>} indicating the filename if present
104      */
105     Optional<String> getFileName();
106 
107     /**
108      * Returns the input stream for this part. This is the content body of the part
109      * and is accessed as a stream to avoid loading potentially large amounts of
110      * data into the heap.
111      *
112      * It is the responsibility of the calling code to close this stream after
113      * receiving it.
114      *
115      * @return an {@code InputStream} representing the content of this part
116      */
117     InputStream getContent();
118 
119     /**
120      * Converts the content stream for this part to the specified class and returns
121      * it. The implementation must convert the stream by finding a
122      * {@link jakarta.ws.rs.ext.MessageBodyReader} that handles the specified type
123      * as well as the {@link MediaType} of the part. If no
124      * {@link jakarta.ws.rs.ext.MessageBodyReader} can be found to perform the
125      * conversion, this method will throw an {@code IllegalArgumentException}.
126      *
127      * The implementation is required to close the content stream when this method
128      * is invoked, so it may only be invoked once. Subsequent invocations will
129      * result in an {@code IllegalStateException}. Likewise this method will throw
130      * an {@code IllegalStateException} if it is called after calling
131      * {@link #getContent} or {@link #getContent(GenericType)}.
132      *
133      * @param type the {@code Class} that the implementation should convert this
134      *             part to
135      * @param <T> the entity type
136      * @return an instance of the specified {@code Class} representing the content
137      *         of this part
138      * @throws IllegalArgumentException if no
139      *                                  {@link jakarta.ws.rs.ext.MessageBodyReader}
140      *                                  can handle the conversion of this part to
141      *                                  the specified type
142      * @throws IllegalStateException    if this method or any of the other
143      *                                  {@code getContent} methods has already been
144      *                                  invoked
145      * @throws IOException              if the
146      *                                  {@link jakarta.ws.rs.ext.MessageBodyReader#readFrom(Class,
147      *                                  java.lang.reflect.Type, java.lang.annotation.Annotation[], MediaType, MultivaluedMap, InputStream)}
148      *                                  method throws an {@code IOException}
149      * @throws WebApplicationException  if the
150      *                                  {@link jakarta.ws.rs.ext.MessageBodyReader#readFrom(Class,
151      *                                  java.lang.reflect.Type, java.lang.annotation.Annotation[], MediaType, MultivaluedMap, InputStream)}
152      *                                  method throws an
153      *                                  {@code WebApplicationException}
154      */
155     //CHECKSTYLE:OFF - More than 3 Exceptions are desired here
156     <T> T getContent(Class<T> type) throws IllegalArgumentException, IllegalStateException, IOException,
157         WebApplicationException;
158     //CHECKSTYLE:ON
159 
160     /**
161      * Converts the content stream for this part to the specified type and returns
162      * it. The implementation must convert the stream by finding a
163      * {@link jakarta.ws.rs.ext.MessageBodyReader} that handles the specified type
164      * as well as the {@link MediaType} of the part. If no
165      * {@link jakarta.ws.rs.ext.MessageBodyReader} can be found to perform the
166      * conversion, this method will throw an {@code IllegalArgumentException}.
167      *
168      * The implementation is required to close the content stream when this method
169      * is invoked, so it may only be invoked once. Subsequent invocations will
170      * result in an {@code IllegalStateException}. Likewise this method will throw
171      * an {@code IllegalStateException} if it is called after calling
172      * {@link #getContent} or {@link #getContent(Class)}.
173      *
174      * @param type the generic type that the implementation should convert this part
175      *             to
176      * @param <T> the entity type
177      * @return an instance of the specified generic type representing the content of
178      *         this part
179      * @throws IllegalArgumentException if no
180      *                                  {@link jakarta.ws.rs.ext.MessageBodyReader}
181      *                                  can handle the conversion of this part to
182      *                                  the specified type
183      * @throws IllegalStateException    if this method or any of the other
184      *                                  {@code getContent} methods has already been
185      *                                  invoked
186      * @throws IOException              if the
187      *                                  {@link jakarta.ws.rs.ext.MessageBodyReader#readFrom(Class,
188      *                                  java.lang.reflect.Type, java.lang.annotation.Annotation[], MediaType, MultivaluedMap, InputStream)}
189      *                                  method throws an {@code IOException}
190      * @throws WebApplicationException  if the
191      *                                  {@link jakarta.ws.rs.ext.MessageBodyReader#readFrom(Class,
192      *                                  java.lang.reflect.Type, java.lang.annotation.Annotation[], MediaType, MultivaluedMap, InputStream)}
193      *                                  method throws an
194      *                                  {@code WebApplicationException}
195      */
196     //CHECKSTYLE:OFF - More than 3 Exceptions are desired here
197     <T> T getContent(GenericType<T> type) throws IllegalArgumentException, IllegalStateException, IOException,
198         WebApplicationException;
199     //CHECKSTYLE:ON
200 
201     /**
202      * Returns an immutable multivalued map of headers for this specific part.
203      *
204      * @return immutable {@code MultivaluedMap<String, String>} of part headers
205      */
206     MultivaluedMap<String, String> getHeaders();
207 
208     /**
209      * Returns the content type of this part, and equivalent to calling
210      * {@code MediaType.valueOf(part.getHeaders().getFirst(HttpHeaders.CONTENT_TYPE))}.
211      *
212      * @return the media type for this part
213      */
214     MediaType getMediaType();
215 
216     /**
217      * Builder for {@link EntityPart} instances.
218      *
219      * @since 3.1
220      */
221     interface Builder {
222 
223         /**
224          * Sets the media type for the EntityPart. This will also set the
225          * {@code Content-Type} header for this part.
226          *
227          * @param mediaType the media type for the part to be built
228          * @return the updated builder
229          * @throws IllegalArgumentException if {@code mediaType} is {@code null}
230          */
231         Builder mediaType(MediaType mediaType) throws IllegalArgumentException;
232 
233         /**
234          * Convenience method for setting the media type for the EntityPart. This will
235          * also set the {@code Content-Type} header for this part. This call is
236          * effectively the same as
237          * {@code mediaType(MediaType.valueOf(mediaTypeString))}.
238          *
239          * @param mediaTypeString the media type for the part to be built
240          * @return the updated builder
241          * @throws IllegalArgumentException if {@code mediaTypeString} cannot be parsed
242          *                                  or is {@code null}
243          */
244         Builder mediaType(String mediaTypeString) throws IllegalArgumentException;
245 
246         /**
247          * Adds a new header or replaces a previously added header and sets the header
248          * value(s).
249          *
250          * @param headerName   the header name
251          * @param headerValues the header value(s)
252          * @return the updated builder
253          * @throws IllegalArgumentException if {@code headerName} is {@code null}
254          */
255         Builder header(String headerName, String... headerValues) throws IllegalArgumentException;
256 
257         /**
258          * Adds new headers or replaces previously added headers. The behavior of this
259          * method would be the same as if iterating over the entry set and invoking the
260          * {@link #header(String, String...)} method.
261          *
262          * @param newHeaders the multivalued map of headers to add to this part
263          * @return the updated builder
264          * @throws IllegalArgumentException if {@code newHeaders} is {@code null}
265          */
266         Builder headers(MultivaluedMap<String, String> newHeaders) throws IllegalArgumentException;
267 
268         /**
269          * Sets the file name for this part. The file name will be specified as an
270          * attribute in the {@code Content-Disposition} header of this part. When this
271          * method is called, the default media type used for the built part will be
272          * "application/octet-stream" if not otherwise specified.
273          *
274          * @param fileName the file name for this part
275          * @return the updated builder
276          * @throws IllegalArgumentException if {@code fileName} is {@code null}
277          */
278         Builder fileName(String fileName) throws IllegalArgumentException;
279 
280         /**
281          * Sets the content for this part. The content of this builder must be specified
282          * before invoking the {@link #build()} method.
283          * <p>
284          * The {@code InputStream} will be closed by the implementation code after
285          * sending the multipart data. Closing the stream before it is sent could result
286          * in unexpected behavior.
287          * </p>
288          *
289          * @param content {@code InputStream} of the content of this part
290          * @return the updated builder
291          * @throws IllegalArgumentException if {@code content} is {@code null}
292          */
293         Builder content(InputStream content) throws IllegalArgumentException;
294 
295         /**
296          * Convenience method, equivalent to calling
297          * {@code fileName(fileName).content(content)}.
298          *
299          * @param fileName the filename of the part.
300          * @param content  the content stream of the part.
301          * @return the updated builder.
302          * @throws IllegalArgumentException if either parameter is {@code null}.
303          */
304         default Builder content(String fileName, InputStream content) throws IllegalArgumentException {
305             return this.fileName(fileName).content(content);
306         }
307 
308         /**
309          * Sets the content for this part. The content of this builder must be specified
310          * before invoking the {@link #build()} method.
311          * <p>
312          * If the content is specified using this method, then the {@link #build()}
313          * method is responsible for finding a registered
314          * {@link jakarta.ws.rs.ext.MessageBodyWriter} that is capable of writing the
315          * object type specified here using the default {@link MediaType} or the
316          * {@link MediaType} specified in the {@link #mediaType(MediaType)} or
317          * {@link #mediaType(String)} methods and using any headers specified via the
318          * {@link #header(String, String...)} or {@link #headers(MultivaluedMap)}
319          * methods.
320          * </p>
321          *
322          * @param content the object to be used as the content
323          * @param type    the type of this object which will be used when selecting the
324          *                appropriate {@link jakarta.ws.rs.ext.MessageBodyWriter}
325          * @param <T>     the entity type
326          * @return the updated builder.
327          * @throws IllegalArgumentException if {@code content} is {@code null}
328          */
329         <T> Builder content(T content, Class<? extends T> type) throws IllegalArgumentException;
330 
331         /**
332          * Sets the content for this part. The content of this builder must be specified
333          * before invoking the {@link #build()} method.
334          * <p>
335          * If the content is specified using this method, then the {@link #build()}
336          * method is responsible for finding a registered
337          * {@link jakarta.ws.rs.ext.MessageBodyWriter} that is capable of writing the
338          * object's class type specified here using the default {@link MediaType} or the
339          * {@link MediaType} specified in the {@link #mediaType(MediaType)} or
340          * {@link #mediaType(String)} methods and using any headers specified via the
341          * {@link #header(String, String...)} or {@link #headers(MultivaluedMap)}
342          * methods.
343          * </p>
344          * <p>
345          * This is the equivalent of calling
346          * {@code content(content, content.getClass())}.
347          * </p>
348          *
349          * @param content the object to be used as the content
350          * @throws IllegalArgumentException if {@code content} is {@code null}
351          * @return the updated builder.
352          */
353         default Builder content(Object content) throws IllegalArgumentException {
354             return this.content(content, content.getClass());
355         }
356 
357         /**
358          * Sets the content for this part. The content of this builder must be specified
359          * before invoking the {@link #build()} method.
360          * <p>
361          * If the content is specified using this method, then the {@link #build()}
362          * method is responsible for finding a registered
363          * {@link jakarta.ws.rs.ext.MessageBodyWriter} that is capable of writing the
364          * object type specified here using the default {@link MediaType} or the
365          * {@link MediaType} specified in the {@link #mediaType(MediaType)} or
366          * {@link #mediaType(String)} methods and using any headers specified via the
367          * {@link #header(String, String...)} or {@link #headers(MultivaluedMap)}
368          * methods.
369          * </p>
370          *
371          * @param content the object to be used as the content
372          * @param type    the generic type of this object which will be used when
373          *                selecting the appropriate
374          *                {@link jakarta.ws.rs.ext.MessageBodyWriter}
375          * @param <T> the entity type
376          * @return the updated builder.
377          * @throws IllegalArgumentException if {@code content} is {@code null}
378          */
379         <T> Builder content(T content, GenericType<T> type) throws IllegalArgumentException;
380 
381         /**
382          * Builds a new EntityPart instance using the provided property values.
383          *
384          * @return {@link EntityPart} instance built from the provided property values.
385          * @throws IllegalStateException   if the content was not specified or no
386          *                                 matching
387          *                                 {@link jakarta.ws.rs.ext.MessageBodyWriter}
388          *                                 was found.
389          * @throws IOException             if the underlying
390          *                                 {@link jakarta.ws.rs.ext.MessageBodyWriter}
391          *                                 throws an {@code IOException}
392          * @throws WebApplicationException if the underlying
393          *                                 {@link jakarta.ws.rs.ext.MessageBodyWriter}
394          *                                 throws a {@code WebApplicationException}
395          */
396         EntityPart build() throws IllegalStateException, IOException, WebApplicationException;
397     }
