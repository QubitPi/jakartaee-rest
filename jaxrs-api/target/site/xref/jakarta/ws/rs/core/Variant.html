19  import java.io.StringWriter;

26  /**
27   * Abstraction for a resource representation variant.
28   *
29   * Contains information about media type, language and encoding of the resource representation.
30   *
31   * @author Paul Sandoz
32   * @author Marc Hadley
33   * @since 1.0
34   */
35  public class Variant {
36  
37      private Locale language;
38      private MediaType mediaType;
39      private String encoding;
40  
41      /**
42       * Create a new instance of Variant.
43       *
44       * @param mediaType the media type of the variant - may be {@code null}.
45       * @param language the language of the variant (two-letter ISO-639 code); may be {@code null}.
46       * @param encoding the content encoding of the variant - may be {@code null}.
47       * @throws java.lang.IllegalArgumentException if all the parameters are {@code null}.
48       * @since 2.0
49       */
50      public Variant(final MediaType mediaType, final String language, final String encoding) {
51          if (mediaType == null && language == null && encoding == null) {
52              throw new IllegalArgumentException("mediaType, language, encoding all null");
53          }
54          this.encoding = encoding;
55          this.language = (language == null) ? null : new Locale(language);
56          this.mediaType = mediaType;
57      }
58  
59      /**
60       * Create a new instance of Variant.
61       *
62       * @param mediaType the media type of the variant - may be {@code null}.
63       * @param language the language of the variant (two-letter ISO-639 code); may be {@code null}.
64       * @param country uppercase two-letter ISO-3166 language code of the variant; may be {@code null} provided
65       * {@code language} is {@code null} too.
66       * @param encoding the content encoding of the variant - may be {@code null}.
67       * @throws java.lang.IllegalArgumentException if all the parameters are {@code null}.
68       * @since 2.0
69       */
70      public Variant(final MediaType mediaType, final String language, final String country, final String encoding) {
71          if (mediaType == null && language == null && encoding == null) {
72              throw new IllegalArgumentException("mediaType, language, encoding all null");
73          }
74          this.encoding = encoding;
75          this.language = (language == null) ? null : new Locale(language, country);
76          this.mediaType = mediaType;
77      }
78  
79      /**
80       * Create a new instance of Variant.
81       *
82       * @param mediaType the media type of the variant - may be {@code null}.
83       * @param language the language of the variant (two-letter ISO-639 code); may be {@code null}.
84       * @param country uppercase two-letter ISO-3166 language code of the variant; may be {@code null} provided
85       * {@code language} is {@code null} too.
86       * @param languageVariant vendor and browser specific language code of the variant (see also {@link Locale} class
87       * description); may be {@code null} provided {@code language} and {@code country} are {@code null} too.
88       * @param encoding the content encoding of the variant - may be {@code null}.
89       * @throws java.lang.IllegalArgumentException if all the parameters are {@code null}.
90       * @since 2.0
91       */
92      public Variant(final MediaType mediaType, final String language, final String country, final String languageVariant, final String encoding) {
93          if (mediaType == null && language == null && encoding == null) {
94              throw new IllegalArgumentException("mediaType, language, encoding all null");
95          }
96          this.encoding = encoding;
97          this.language = (language == null) ? null : new Locale(language, country, languageVariant);
98          this.mediaType = mediaType;
99      }

101     /**
102      * Create a new instance of Variant.
103      *
104      * @param mediaType the media type of the variant - may be {@code null}.
105      * @param language the language of the variant - may be {@code null}.
106      * @param encoding the content encoding of the variant - may be {@code null}.
107      * @throws java.lang.IllegalArgumentException if all the parameters are {@code null}.
108      */
109     public Variant(final MediaType mediaType, final Locale language, final String encoding) {
110         if (mediaType == null && language == null && encoding == null) {
111             throw new IllegalArgumentException("mediaType, language, encoding all null");
112         }
113         this.encoding = encoding;
114         this.language = language;
115         this.mediaType = mediaType;
116     }

155     /**
156      * Create a {@link VariantListBuilder} initialized with a set of supported media types.

195     /**
196      * Generate hash code from variant properties.
197      *
198      * @return the hash code.
199      */
200     @Override
201     public int hashCode() {
202         return Objects.hash(this.language, this.mediaType, this.encoding);
203     }
204 
205     /**
206      * Compares obj to this variant to see if they are the same considering all property values.
207      *
208      * @param obj the object to compare to.
209      * @return true if the two variants are the same, false otherwise.
210      */
211     @Override
212     public boolean equals(final Object obj) {
213         if (obj == null) {
214             return false;
215         }
216         if (getClass() != obj.getClass()) {
217             return false;
218         }
219         final Variant other = (Variant) obj;
220         if (!Objects.equals(this.language, other.language)) {
221             return false;
222         }
223         if (!Objects.equals(this.mediaType, other.mediaType)) {
224             return false;
225         }
226         return Objects.equals(this.encoding, other.encoding);
227     }

229     @Override
230     public String toString() {
231         StringWriter w = new StringWriter();
232         w.append("Variant[mediaType=");
233         w.append(mediaType == null ? "null" : mediaType.toString());
234         w.append(", language=");
235         w.append(language == null ? "null" : language.toString());
236         w.append(", encoding=");
237         w.append(encoding == null ? "null" : encoding);
238         w.append("]");
239         return w.toString();
240     }

242     /**
243      * A builder for a list of representation variants.
244      */
245     public abstract static class VariantListBuilder {
246 
247         /**
248          * Protected constructor, use the static {@code newInstance} method to obtain an instance.
249          */
250         protected VariantListBuilder() {
251         }
252 
253         /**
254          * Create a new builder instance.
255          *
256          * @return a new builder instance.
257          */
258         public static VariantListBuilder newInstance() {
259             return RuntimeDelegate.getInstance().createVariantListBuilder();
260         }
261 
262         /**
263          * Add the current combination of metadata to the list of supported variants (provided the current combination of
264          * metadata is not empty) and build a list of representation variants from the current state of the builder. After this
265          * method is called the builder is reset to an empty state.
266          *
267          * @return a list of representation variants.
268          */
269         public abstract List<Variant> build();
270 
271         /**
272          * Add the current combination of metadata to the list of supported variants, after this method is called the current
273          * combination of metadata is emptied.
274          * <p>
275          * If more than one value is supplied for one or more of the variant properties then a variant will be generated for
276          * each possible combination. E.g. in the following {@code list} would have five (4 + 1) members:
277          * </p>
278          *
279          * <pre>
280          * List&lt;Variant&gt; list = VariantListBuilder.newInstance()
281          *         .languages(Locale.ENGLISH, Locale.FRENCH).encodings("zip", "identity").add()
282          *         .languages(Locale.GERMAN).mediaTypes(MediaType.TEXT_PLAIN_TYPE).add()
283          *         .build()
284          * </pre>
285          * <p>
286          * Note that it is not necessary to call the {@code add()} method immediately before the build method is called. E.g.
287          * the resulting list produced in the example above would be identical to the list produced by the following code:
288          * </p>
289          *
290          * <pre>
291          * List&lt;Variant&gt; list = VariantListBuilder.newInstance()
292          *         .languages(Locale.ENGLISH, Locale.FRENCH).encodings("zip", "identity").add()
293          *         .languages(Locale.GERMAN).mediaTypes(MediaType.TEXT_PLAIN_TYPE)
294          *         .build()
295          * </pre>
296          *
297          * @return the updated builder.
298          * @throws IllegalStateException if there is not at least one mediaType, language or encoding set for the current
299          * variant.
300          */
301         public abstract VariantListBuilder add();
302 
303         /**
304          * Set the language(s) for this variant.
305          *
306          * @param languages the available languages.
307          * @return the updated builder.
308          */
309         public abstract VariantListBuilder languages(Locale... languages);
310 
311         /**
312          * Set the encoding(s) for this variant.
313          *
314          * @param encodings the available encodings.
315          * @return the updated builder.
316          */
317         public abstract VariantListBuilder encodings(String... encodings);
318 
319         /**
320          * Set the media type(s) for this variant.
321          *
322          * @param mediaTypes the available mediaTypes. If specific charsets are supported they should be included as parameters
323          * of the respective media type.
324          * @return the updated builder.
325          */
326         public abstract VariantListBuilder mediaTypes(MediaType... mediaTypes);
327     }
