17  package jakarta.ws.rs.core;
18  
19  import java.util.Date;
20  import java.util.List;
21  
22  import jakarta.ws.rs.core.Response.ResponseBuilder;
23  
24  /**
25   * An injectable helper for request processing, all methods throw an {@link IllegalStateException} if called outside the
26   * scope of a request (e.g. from a provider constructor).
27   *
28   * Precondition processing (see the {@code evaluatePreconditions} methods) can result in either a {@code null} return
29   * value to indicate that preconditions have been met and that the request should continue, or a non-{@code null} return
30   * value to indicate that preconditions were not met. In the event that preconditions were not met, the returned
31   * {@code ResponseBuilder} instance will have an appropriate status and will also include a {@code Vary} header if the
32   * {@link #selectVariant(List)} method was called prior to to calling {@code evaluatePreconditions}. It is the
33   * responsibility of the caller to check the status and add additional metadata if required. E.g., see
34   * <a href="http://www.w3.org/Protocols/rfc2616/rfc2616-sec10.html#sec10.3.5">HTTP/1.1, section 10.3.5</a> for details
35   * of the headers that are expected to accompany a {@code 304 Not Modified} response.
36   *
37   * @author Paul Sandoz
38   * @author Marc Hadley
39   * @author Marek Potociar
40   * @since 1.0
41   */
42  public interface Request {
43  
44      /**
45       * Get the request method, e.g. GET, POST, etc.
46       *
47       * @return the request method.
48       * @see jakarta.ws.rs.HttpMethod
49       */
50      public String getMethod();
51  
52      /**
53       * Select the representation variant that best matches the request. Returns {@code null} in case there is no matching
54       * variant in the list.
55       * <p>
56       * More explicit variants are chosen ahead of less explicit ones. A vary header is computed from the supplied list and
57       * automatically added to the response.
58       * </p>
59       *
60       * @param variants a list of Variant that describe all of the available representation variants.
61       * @return the variant that best matches the request or {@code null} if there's no match.
62       * @throws java.lang.IllegalArgumentException if variants is empty or {@code null}.
63       * @throws java.lang.IllegalStateException if called outside the scope of a request.
64       * @see Variant.VariantListBuilder
65       */
66      public Variant selectVariant(List<Variant> variants);
67  
68      /**
69       * Evaluate request preconditions based on the passed in value.
70       *
71       * @param eTag an ETag for the current state of the resource
72       * @return {@code null} if the preconditions are met or a {@code ResponseBuilder} set with the appropriate status if the
73       * preconditions are not met. A returned {@code ResponseBuilder} will include an ETag header set with the value of eTag,
74       * provided none of the precondition evaluation has failed, in which case the ETag header would not be included and the
75       * status code of the returned {@code ResponseBuilder} would be set to {@link Response.Status#PRECONDITION_FAILED}.
76       * @throws java.lang.IllegalArgumentException if eTag is {@code null}.
77       * @throws java.lang.IllegalStateException if called outside the scope of a request.
78       */
79      public ResponseBuilder evaluatePreconditions(EntityTag eTag);
80  
81      /**
82       * Evaluate request preconditions based on the passed in value.
83       *
84       * @param lastModified a date that specifies the modification date of the resource
85       * @return {@code null} if the preconditions are met or a {@code ResponseBuilder} set with the appropriate status if the
86       * preconditions are not met.
87       * @throws java.lang.IllegalArgumentException if lastModified is {@code null}.
88       * @throws java.lang.IllegalStateException if called outside the scope of a request.
89       */
90      public ResponseBuilder evaluatePreconditions(Date lastModified);
91  
92      /**
93       * Evaluate request preconditions based on the passed in value.
94       *
95       * @param lastModified a date that specifies the modification date of the resource
96       * @param eTag an ETag for the current state of the resource
97       * @return {@code null} if the preconditions are met or a {@code ResponseBuilder} set with the appropriate status if the
98       * preconditions are not met. A returned {@code ResponseBuilder} will include an ETag header set with the value of eTag,
99       * provided none of the precondition evaluation has failed, in which case the ETag header would not be included and the
100      * status code of the returned {@code ResponseBuilder} would be set to {@link Response.Status#PRECONDITION_FAILED}.
101      * @throws java.lang.IllegalArgumentException if lastModified or eTag is {@code null}.
102      * @throws java.lang.IllegalStateException if called outside the scope of a request.
103      */
104     public ResponseBuilder evaluatePreconditions(Date lastModified, EntityTag eTag);
105 
106     /**
107      * Evaluate request preconditions for a resource that does not currently exist. The primary use of this method is to
108      * support the <a href="https://tools.ietf.org/html/rfc7232#section-3.1"> If-Match: *</a> and
109      * <a href="https://tools.ietf.org/html/rfc7232#section-3.2"> If-None-Match: *</a> preconditions.
110      *
111      * <p>
112      * Note that precondition <code>If-None-Match: <i>something</i></code> will never be
113      * considered to have been met, and it is the application's responsibility to enforce any additional method-specific
114      * semantics. E.g. a {@code PUT} on a resource that does not exist might succeed whereas a {@code GET} on a resource
115      * that does not exist would likely result in a 404 response. It would be the responsibility of the application to
116      * generate the 404 response.
117      * </p>
118      *
119      * @return {@code null} if the preconditions are met or a {@code ResponseBuilder} set with the appropriate status if the
120      * preconditions are not met.
121      * @throws IllegalStateException if called outside the scope of a request.
122      * @since 1.1
123      */
124     public ResponseBuilder evaluatePreconditions();
125 
126 }
