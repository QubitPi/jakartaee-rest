17  package jakarta.ws.rs;
18  
19  import java.lang.annotation.Documented;
20  import java.lang.annotation.ElementType;
21  import java.lang.annotation.Retention;
22  import java.lang.annotation.RetentionPolicy;
23  import java.lang.annotation.Target;
24  
25  /**
26   * Meta-annotation used to create name binding annotations for filters and interceptors.
27   *
28   * Name binding via annotations is only supported as part of the Server API. In name binding, a <i>name-binding</i>
29   * annotation is first defined using the {@code @NameBinding} meta-annotation:
30   *
31   * <pre>
32   *  &#64;Target({ ElementType.TYPE, ElementType.METHOD })
33   *  &#64;Retention(value = RetentionPolicy.RUNTIME)
34   *  <b>&#64;NameBinding</b>
35   *  <b>public @interface Logged</b> { }
36   * </pre>
37   *
38   * The defined name-binding annotation is then used to decorate a filter or interceptor class (more than one filter or
39   * interceptor may be decorated with the same name-binding annotation):
40   *
41   * <pre>
42   *  <b>&#64;Logged</b>
43   *  public class LoggingFilter
44   *          implements ContainerRequestFilter, ContainerResponseFilter {
45   *      ...
46   *  }
47   * </pre>
48   *
49   * At last, the name-binding annotation is applied to the resource method(s) to which the name-bound JAX-RS provider(s)
50   * should be bound to:
51   *
52   * <pre>
53   *  &#64;Path("/")
54   *  public class MyResourceClass {
55   *      &#64;GET
56   *      &#64;Produces("text/plain")
57   *      &#64;Path("{name}")
58   *      <b>&#64;Logged</b>
59   *      public String hello(@PathParam("name") String name) {
60   *          return "Hello " + name;
61   *      }
62   *  }
63   * </pre>
64   *
65   * A name-binding annotation may also be attached to a custom JAX-RS {@link jakarta.ws.rs.core.Application} subclass. In
66   * such case a name-bound JAX-RS provider bound by the annotation will be applied to all {@link HttpMethod resource and
67   * sub-resource methods} in the JAX-RS application:
68   *
69   * <pre>
70   *  <b>&#64;Logged</b>
71   *  &#64;ApplicationPath("myApp")
72   *  public class MyApplication extends jakarta.ws.rs.core.Application {
73   *      ...
74   *  }
75   * </pre>
76   *
77   * @author Santiago Pericas-Geertsen
78   * @author Marek Potociar
79   * @since 2.0
80   */
81  @Target(ElementType.ANNOTATION_TYPE)
82  @Retention(RetentionPolicy.RUNTIME)
83  @Documented
84  public @interface NameBinding {
85  }
