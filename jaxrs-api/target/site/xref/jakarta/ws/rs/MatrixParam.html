17  package jakarta.ws.rs;
18  
19  import java.lang.annotation.Documented;
20  import java.lang.annotation.ElementType;
21  import java.lang.annotation.Retention;
22  import java.lang.annotation.RetentionPolicy;
23  import java.lang.annotation.Target;
24  
25  /**
26   * Binds the value(s) of a URI matrix parameter to a resource method parameter, resource class field, or resource class
27   * bean property. Values are URL decoded unless this is disabled using the {@link Encoded} annotation. A default value
28   * can be specified using the {@link DefaultValue} annotation.
29   * <p>
30   * Note that the {@code @MatrixParam} {@link #value() annotation value} refers to a name of a matrix parameter that
31   * resides in the last matched path segment of the {@link Path}-annotated Java structure that injects the value of the
32   * matrix parameter.
33   * </p>
34   * <p>
35   * The type {@code T} of the annotated parameter, field or property must either:
36   * </p>
37   * <ol>
38   * <li>Be a primitive type</li>
39   * <li>Have a constructor that accepts a single {@code String} argument</li>
40   * <li>Have a static method named {@code valueOf} or {@code fromString} that accepts a single {@code String} argument
41   * (see, for example, {@link Integer#valueOf(String)})</li>
42   * <li>Have a registered implementation of {@link jakarta.ws.rs.ext.ParamConverterProvider} JAX-RS extension SPI that
43   * returns a {@link jakarta.ws.rs.ext.ParamConverter} instance capable of a "from string" conversion for the type.</li>
44   * <li>Be {@code List<T>}, {@code Set<T>}, {@code SortedSet<T>} or {@code T[]} array, where {@code T} satisfies 2, 3 or
45   * 4 above. The resulting collection is read-only.</li>
46   * </ol>
47   *
48   * <p>
49   * If the type is not one of the collection types listed in 5 above and the matrix parameter is represented by multiple
50   * values then the first value (lexically) of the parameter is used.
51   * </p>
52   *
53   * <p>
54   * Because injection occurs at object creation time, use of this annotation on resource class fields and bean properties
55   * is only supported for the default per-request resource class lifecycle. Resource classes using other lifecycles
56   * should only use this annotation on resource method parameters.
57   * </p>
58   *
59   * @author Paul Sandoz
60   * @author Marc Hadley
61   * @see DefaultValue
62   * @see Encoded
63   * @see <a href="http://www.w3.org/DesignIssues/MatrixURIs.html">Matrix URIs</a>
64   * @since 1.0
65   */
66  @Target({ ElementType.PARAMETER, ElementType.METHOD, ElementType.FIELD })
67  @Retention(RetentionPolicy.RUNTIME)
68  @Documented
69  public @interface MatrixParam {
70  
71      /**
72       * Defines the name of the URI matrix parameter whose value will be used to initialize the value of the annotated method
73       * argument, class field or bean property. The name is specified in decoded form, any percent encoded literals within
74       * the value will not be decoded and will instead be treated as literal text. E.g. if the parameter name is "a b" then
75       * the value of the annotation is "a b", <i>not</i> "a+b" or "a%20b".
76       *
77       * @return URI matrix parameter name.
78       */
79      String value();
80  }
