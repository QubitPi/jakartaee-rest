17  package jakarta.ws.rs;
18  
19  /**
20   * A base JAX-RS runtime processing exception.
21   *
22   * The exception of this type is thrown during HTTP request or response processing, to signal a runtime processing
23   * failure. Typical classes of failures covered by {@code ProcessingException} include
24   * <ul>
25   * <li>failures in filter or interceptor chain execution,</li>
26   * <li>errors caused by missing message body readers or writers for the particular Java type and media type
27   * combinations,</li>
28   * <li>propagated {@link java.io.IOException IO exceptions} thrown by entity {@link jakarta.ws.rs.ext.MessageBodyReader
29   * readers} and {@link jakarta.ws.rs.ext.MessageBodyWriter writers} during entity serialization and de-serialization.</li>
30   * </ul>
31   * as well as any other JAX-RS runtime processing errors. The exception message or nested {@link Throwable} cause SHOULD
32   * contain additional information about the reason of the processing failure.
33   * <p>
34   * Note that the exception is used to indicate (internal) JAX-RS processing errors. It is not used to indicate HTTP
35   * error response states. A HTTP error response is represented by a {@link jakarta.ws.rs.WebApplicationException} class or
36   * one of it's sub-classes.
37   * </p>
38   *
39   * @author Marek Potociar
40   * @since 2.0
41   */
42  public class ProcessingException extends RuntimeException {
43  
44      private static final long serialVersionUID = -4232431597816056514L;
45  
46      /**
47       * Constructs a new JAX-RS runtime processing exception with the specified cause and a detail message of
48       * {@code (cause==null ? null : cause.toString())} (which typically contains the class and detail message of
49       * {@code cause}). This constructor is useful for runtime exceptions that are little more than wrappers for other
50       * throwables.
51       *
52       * @param cause the cause (which is saved for later retrieval by the {@link #getCause()} method). (A {@code null} value
53       * is permitted, and indicates that the cause is nonexistent or unknown.)
54       */
55      public ProcessingException(final Throwable cause) {
56          super(cause);
57      }
58  
59      /**
60       * <p>
61       * Constructs a new JAX-RS runtime processing exception with the specified detail message and cause.
62       * </p>
63       * Note that the detail message associated with {@code cause} is <i>not</i> automatically incorporated in this runtime
64       * exception's detail message.
65       *
66       * @param message the detail message (which is saved for later retrieval by the {@link #getMessage()} method).
67       * @param cause the cause (which is saved for later retrieval by the {@link #getCause()} method). (A {@code null} value
68       * is permitted, and indicates that the cause is nonexistent or unknown.)
69       */
70      public ProcessingException(final String message, final Throwable cause) {
71          super(message, cause);
72      }
73  
74      /**
75       * Constructs a new JAX-RS runtime processing exception with the specified detail message. The cause is not initialized,
76       * and may subsequently be initialized by a call to {@link #initCause}.
77       *
78       * @param message the detail message (which is saved for later retrieval by the {@link #getMessage()} method).
79       */
80      public ProcessingException(final String message) {
81          super(message);
82      }
83  }
