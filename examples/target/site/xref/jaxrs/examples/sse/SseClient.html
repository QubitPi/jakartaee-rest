11  package jaxrs.examples.sse;
12  
13  import jakarta.ws.rs.client.ClientBuilder;
14  import jakarta.ws.rs.client.Entity;
15  import jakarta.ws.rs.client.WebTarget;
16  import jakarta.ws.rs.sse.SseEventSource;
17  
18  /**
19   * Examples of Client-side Server-sent events processing.
20   *
21   * @author Marek Potociar (marek.potociar at oracle.com)
22   */
23  public final class SseClient {
24  
25      static final WebTarget TARGET = ClientBuilder.newClient().target("server-sent-events");
26  
27      private SseClient() {
28      }
29  
30      public static void main(String[] args) {
31          consumeAllEvents();
32      }
33  
34      private static void consumeAllEvents() {
35  
36          // EventSource#register(Consumer<InboundSseEvent>)
37          // consumes all events, writes then on standard out (System.out::println)
38          try (final SseEventSource eventSource = SseEventSource.target(TARGET)
39                  .build()) {
40  
41              eventSource.register(System.out::println);
42              eventSource.open();
43  
44              for (int counter = 0; counter < 5; counter++) {
45                  TARGET.request().post(Entity.text("message " + counter));
46              }
47  
48              Thread.sleep(500); // make sure all the events have time to arrive
49          } catch (InterruptedException e) {
50              e.printStackTrace();
51          }
52  
53          // EventSource#register(Consumer<InboundSseEvent>, Consumer<Throwable>)
54          // consumes all events and all exceptions, writing both on standard out.
55          try (final SseEventSource eventSource = SseEventSource.target(TARGET).build()) {
56  
57              eventSource.register(System.out::println, Throwable::printStackTrace);
58              eventSource.open();
59  
60              for (int counter = 0; counter < 5; counter++) {
61                  TARGET.request().post(Entity.text("message " + counter));
62              }
63  
64              Thread.sleep(500); // make sure all the events have time to arrive
65          } catch (InterruptedException e) {
66              e.printStackTrace();
67          }
68  
69          // EventSource#register(Consumer<InboundSseEvent>, Consumer<Throwable>, Runnable)
70          // consumes all events and all exceptions, writing both on standard out.
71          // registers onComplete callback, which will print out a message "There will be no further events."
72          try (final SseEventSource eventSource = SseEventSource.target(TARGET).build()) {
73  
74              eventSource.register(
75                      System.out::println,
76                      Throwable::printStackTrace,
77                      () -> System.out.println("There will be no further events."));
78              eventSource.open();
79  
80              for (int counter = 0; counter < 5; counter++) {
81                  TARGET.request().post(Entity.text("message " + counter));
82              }
83  
84              Thread.sleep(500); // make sure all the events have time to arrive
85          } catch (InterruptedException e) {
86              e.printStackTrace();
87          }
88      }
89  }
