36      @XmlRootElement
37      public static class Customer {

39          private final String name;
40  
41          public Customer(String name) {
42              this.name = name;
43          }
44  
45          public String getName() {
46              return name;
47          }
48      }

50      public void clientBootstrapping() {
51          // Default client instantiation using default configuration
52          Client defaultClient = ClientBuilder.newClient();
53          defaultClient.property("CUSTOM_PROPERTY", "CUSTOM_VALUE");
54          assert defaultClient != null;
55  
56          // Default client instantiation using custom configuration
57  
58          Client defaultConfiguredClient = ClientBuilder.newClient(defaultClient.getConfiguration());
59          assert defaultConfiguredClient != null;
60  
61          ///////////////////////////////////////////////////////////
62  
63          // Custom client instantiation examples
64          ThrottledClient myClient = new ThrottledClient();
65          assert myClient != null;
66  
67          ThrottledClient myConfiguredClient = new ThrottledClient(10);
68          assert myConfiguredClient != null;
69      }
70  
71      public void fluentMethodChaining() {
72          Client client = ClientBuilder.newClient();
73          Response res = client.target("http://example.org/hello")
74                  .request("text/plain").get();
75  
76          Response res2 = client.target("http://example.org/hello")
77                  .queryParam("MyParam", "...")
78                  .request("text/plain")
79                  .header("MyHeader", "...")
80                  .get();
81      }
82  
83      public void typeRelationships() {
84          Client client = ClientBuilder.newClient();
85          WebTarget uri = client.target("");
86          Invocation.Builder builder = uri.request("text/plain");
87  
88          SyncInvoker syncInvoker = builder;
89          AsyncInvoker asyncInvoker = builder.async();
90          Invocation inv = builder.buildGet();
91  
92          Response r1 = builder.get();
93          Response r2 = syncInvoker.get();
94          Response r3 = inv.invoke();
95  
96          Future<Response> fr1 = asyncInvoker.get();
97          Future<Response> fr2 = inv.submit();
98      }
99  
100     public void benefitsOfResourceUri() {
101         Client client = ClientBuilder.newClient();
102         WebTarget base = client.target("http://example.org/");
103         WebTarget hello = base.path("hello").path("{whom}");
104         final WebTarget whomToGreet = hello.resolveTemplate("whom", "world");
105         Response res = whomToGreet.request().get();
106     }
107 
108     public void gettingAndPostingCustomers() {
109         Client client = ClientBuilder.newClient();
110         Customer c = client.target("http://examples.org/customers/123")
111                 .request("application/xml").get(Customer.class);
112         Response res = client.target("http://examples.org/premium-customers/")
113                 .request().post(entity(c, "application/xml"));
114     }
115 
116     public void asyncSamples() throws Exception {
117         Client client = ClientBuilder.newClient();
118         Future<Customer> fc = client.target("http://examples.org/customers/123")
119                 .request("application/xml").async().get(Customer.class);
120         Customer c = fc.get();
121     }
