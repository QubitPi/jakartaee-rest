11  package jaxrs.examples.client.webdav;
12  
13  import jakarta.ws.rs.core.Feature;
14  import jakarta.ws.rs.core.FeatureContext;
15  
16  /**
17   * Test of WebDAV Client API Extension.
18   *
19   * @author Marek Potociar
20   */
21  public class WebDavClientTest {
22  
23      private WebDavClient createClient() {
24          return new WebDavClient();
25      }
26  
27      static class TestFeature implements Feature {
28  
29          @Override
30          public boolean configure(FeatureContext context) {
31              throw new UnsupportedOperationException("Not supported yet.");
32          }
33      }
34  
35      public void fluentUseCases() {
36          WebDavClient client = createClient();
37  
38          TestFeature testFeature = new TestFeature();
39          client.register(testFeature);
40  
41          client.target("http://examples.jaxrs.com/webdav/");
42          client.target("http://examples.jaxrs.com/webdav/").register(testFeature);
43          client.target("http://examples.jaxrs.com/webdav/").request().property("foo", "bar");
44          client.target("http://examples.jaxrs.com/webdav/").request().buildGet().property("foo", "bar");
45  
46          // HTTP
47          client.target("http://examples.jaxrs.com/webdav/").request().get();
48          client.target("http://examples.jaxrs.com/webdav/").request().async().get();
49          client.target("http://examples.jaxrs.com/webdav/").request().buildGet().invoke();
50          client.target("http://examples.jaxrs.com/webdav/").request().buildGet().submit();
51          // WebDAV
52          client.target("http://examples.jaxrs.com/webdav/").request().search(null);
53          client.target("http://examples.jaxrs.com/webdav/").request().async().search(null);
54          client.target("http://examples.jaxrs.com/webdav/").request().buildSearch(null).invoke();
55          client.target("http://examples.jaxrs.com/webdav/").request().buildSearch(null).submit();
56  
57          // HTTP
58          client.target("http://examples.jaxrs.com/webdav/").path("123").request().get();
59          client.target("http://examples.jaxrs.com/webdav/").path("123").request().async().get();
60          client.target("http://examples.jaxrs.com/webdav/").path("123").request().buildGet().invoke();
61          client.target("http://examples.jaxrs.com/webdav/").path("123").request().buildGet().submit();
62          // WebDAV
63          client.target("http://examples.jaxrs.com/webdav/").path("123").request().search(null);
64          client.target("http://examples.jaxrs.com/webdav/").path("123").request().async().search(null);
65          client.target("http://examples.jaxrs.com/webdav/").path("123").request().buildSearch(null).invoke();
66          client.target("http://examples.jaxrs.com/webdav/").path("123").request().buildSearch(null).submit();
67  
68          // HTTP
69          client.target("http://examples.jaxrs.com/webdav/").path("123").request("text/plain").get();
70          client.target("http://examples.jaxrs.com/webdav/").path("123").request("text/plain").async().get();
71          client.target("http://examples.jaxrs.com/webdav/").path("123").request("text/plain").buildGet().invoke();
72          client.target("http://examples.jaxrs.com/webdav/").path("123").request("text/plain").buildGet().submit();
73          // WebDAV
74          client.target("http://examples.jaxrs.com/webdav/").path("123").request("text/plain").search(null);
75          client.target("http://examples.jaxrs.com/webdav/").path("123").request("text/plain").async().search(null);
76          client.target("http://examples.jaxrs.com/webdav/").path("123").request("text/plain").buildSearch(null).invoke();
77          client.target("http://examples.jaxrs.com/webdav/").path("123").request("text/plain").buildSearch(null).submit();
78  
79          // HTTP
80          client.target("http://examples.jaxrs.com/webdav/").path("123").request("text/plain").header("custom-name", "custom_value").get();
81          client.target("http://examples.jaxrs.com/webdav/").path("123").request("text/plain").header("custom-name", "custom_value").async().get();
82          client.target("http://examples.jaxrs.com/webdav/").path("123").request("text/plain").header("custom-name", "custom_value").buildGet().invoke();
83          client.target("http://examples.jaxrs.com/webdav/").path("123").request("text/plain").header("custom-name", "custom_value").buildGet().submit();
84          // WebDAV
85          client.target("http://examples.jaxrs.com/webdav/").path("123").request("text/plain").header("custom-name", "custom_value").search(null);
86          client.target("http://examples.jaxrs.com/webdav/").path("123").request("text/plain").header("custom-name", "custom_value").async().search(null);
87          client.target("http://examples.jaxrs.com/webdav/").path("123").request("text/plain").header("custom-name", "custom_value").buildSearch(null).invoke();
88          client.target("http://examples.jaxrs.com/webdav/").path("123").request("text/plain").header("custom-name", "custom_value").buildSearch(null).submit();
89      }
90  }
