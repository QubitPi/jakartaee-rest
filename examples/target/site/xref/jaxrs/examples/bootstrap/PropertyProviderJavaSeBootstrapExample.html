17  package jaxrs.examples.bootstrap;
18  
19  import java.net.URI;
20  
21  import jakarta.ws.rs.SeBootstrap;
22  import jakarta.ws.rs.core.Application;
23  import org.eclipse.microprofile.config.Config;
24  import org.eclipse.microprofile.config.ConfigProvider;
25  
26  /**
27   * Java SE bootstrap example utilizing a property provider.
28   * <p>
29   * This example demonstrates bootstrapping on Java SE platforms using values explicitly bulk-loaded from a property
30   * provider plus statically overridden properties. In particular, this demo first uses Eclipse Microprofile Config as a
31   * property provider for all properties supported by the implementation, but then explicitly enforces HTTPS within the
32   * bootstrap code.
33   * </p>
34   * <p>
35   * To actually run this example, an implementation of Eclipse Microprofile Config has to be added to the classpath, and
36   * the configuration options to be customized have to be provided, e. g. as environment variables:
37   * </p>
38   *
39   * <pre>
40   * export jakarta.ws.rs.SeBootstrap.Port=8888;
41   * java -Djakarta.ws.rs.SeBootstrap.Host=127.0.0.1 PropertyProviderJavaSeBootstrapExample;
42   * </pre>
43   * <p>
44   * Thanks to the <em>explicit</em> use of Microprofile Config in the application, this example is completely portable.
45   * </p>
46   * <p>
47   * This example uses some basic <em>external</em> JSSE configuration:
48   * </p>
49   * <ul>
50   * <li>{@code javax.net.ssl.keyStore=~/.keystore} - HTTPS: Path to a keystore holding an X.509 certificate for
51   * {@code CN=localhost}</li>
52   * <li>{@code javax.net.ssl.keyStorePassword=...} - HTTPS: Password of that keystore</li>
53   * </ul>
54   *
55   * @author Markus KARG (markus@headcrashing.eu)
56   * @since 3.1
57   */
58  public final class PropertyProviderJavaSeBootstrapExample {
59  
60      private PropertyProviderJavaSeBootstrapExample() {
61      }
62  
63      /**
64       * Runs this example.
65       *
66       * @param args unused command line arguments
67       * @throws InterruptedException when process is killed
68       */
69      public static void main(final String[] args) throws InterruptedException {
70          final Application application = new HelloWorld();
71  
72          final Config config = ConfigProvider.getConfig();
73  
74          final SeBootstrap.Configuration requestedConfiguration = SeBootstrap.Configuration.builder().from(config::getOptionalValue)
75                  .protocol("HTTPS").build();
76  
77          SeBootstrap.start(application, requestedConfiguration).thenAccept(instance -> {
78              instance.stopOnShutdown(stopResult ->
79                      System.out.printf("Stop result: %s [Native stop result: %s].%n", stopResult,
80                              stopResult.unwrap(Object.class)));
81              final URI uri = instance.configuration().baseUri();
82              System.out.printf("Instance %s running at %s [Native handle: %s].%n", instance, uri,
83                      instance.unwrap(Object.class));
84              System.out.println("Send SIGKILL to shutdown.");
85          });
86  
87          Thread.currentThread().join();
88      }
89  
90  }
