17  package jaxrs.examples.bootstrap;
18  
19  import java.net.URI;
20  
21  import jakarta.ws.rs.SeBootstrap;
22  import jakarta.ws.rs.core.Application;
23  
24  /**
25   * Java SE bootstrap example demonstrating the use of native properties.
26   * <p>
27   * This example demonstrates bootstrapping on Java SE platforms with native properties, in particular by explicitly
28   * selecting the Grizzly2 backend in Jersey. It will effectively startup the {@link HelloWorld} application at the URL
29   * {@code http://localhost:80/}, i. e. the Grizzly2 backend selects exactly port 80 as this is its default IP port. The
30   * actual configuration needs to be queried after bootstrapping, otherwise callers would be unaware of the actual chosen
31   * port, as Grizzly2's default behavior is not publicly documented.
32   * </p>
33   * <p>
34   * This is a native example, hence it only works with Jersey, and in particular only with its Grizzly2 backend. To
35   * actually run this demo, the following libraries have to be found on the classpath:
36   * </p>
37   * <ul>
38   * <li>jersey-server</li>
39   * <li>jersey-container-grizzly2-http</li>
40   * </ul>
41   *
42   * @author Markus KARG (markus@headcrashing.eu)
43   * @since 3.1
44   */
45  public final class NativeJavaSeBootstrapExample {
46  
47      private NativeJavaSeBootstrapExample() {
48      }
49  
50      /**
51       * Runs this example.
52       *
53       * @param args unused command line arguments
54       * @throws InterruptedException when process is killed
55       * @throws ClassNotFoundException when Jersey's Grizzly backend is not on the classpath
56       */
57      public static void main(final String[] args) throws InterruptedException, ClassNotFoundException {
58          final Application application = new HelloWorld();
59  
60          final SeBootstrap.Configuration requestedConfiguration = SeBootstrap.Configuration.builder()
61                  .property("jersey.config.server.httpServerClass",
62                          Class.forName("org.glassfish.jersey.grizzly2.httpserver.GrizzlyHttpServer"))
63                  .build();
64  
65          SeBootstrap.start(application, requestedConfiguration).thenAccept(instance -> {
66              instance.stopOnShutdown(stopResult ->
67                      System.out.printf("Stop result: %s [Native stop result: %s].%n", stopResult,
68                              stopResult.unwrap(Object.class)));
69              final URI uri = instance.configuration().baseUri();
70              System.out.printf("Instance %s running at %s [Native handle: %s].%n", instance, uri,
71                      instance.unwrap(Object.class));
72              System.out.println("Send SIGKILL to shutdown.");
73          });
74  
75          Thread.currentThread().join();
76      }
77  
78  }
