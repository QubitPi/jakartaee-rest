11  package jaxrs.examples.link.clusterservice;
12  
13  import jakarta.ws.rs.client.Client;
14  import jakarta.ws.rs.client.ClientBuilder;
15  import jakarta.ws.rs.core.Link;
16  import jakarta.ws.rs.core.Response;
17  
18  /**
19   * ClientTest class.
20   *
21   * @author Santiago.Pericas-Geertsen@oracle.com
22   */
23  public class ClientTest {
24  
25      public void test() {
26          Client client = ClientBuilder.newClient();
27  
28          // Get cluster representation -- entry point
29          Response rc = client.target("/cluster").request("application/json").get();
30  
31          // Ensure cluster is online
32          if (rc.hasLink("onliner")) {
33              client.invocation(rc.getLink("onliner")).buildPost(null).invoke();
34          }
35  
36          // Start all machines in cluster
37          Cluster c = rc.readEntity(Cluster.class);
38          for (Machine m : c.getMachines()) {
39              // Machine name is need for URI template in link
40              Link l = rc.getLinkBuilder("item").build(m.getName());
41  
42              // Create invocation from link and call invoke()
43              Response rm = client.invocation(l).buildGet().invoke();
44  
45              // Start machine if not started already
46              if (rm.hasLink("starter")) {
47                  client.invocation(rm.getLink("starter")).buildPost(null).invoke();
48              }
49          }
50      }
51  }
