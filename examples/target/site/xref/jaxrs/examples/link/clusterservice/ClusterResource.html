11  package jaxrs.examples.link.clusterservice;
12  
13  import java.net.URI;
14  
15  import jakarta.ws.rs.GET;
16  import jakarta.ws.rs.POST;
17  import jakarta.ws.rs.Path;
18  import jakarta.ws.rs.Produces;
19  import jakarta.ws.rs.core.Context;
20  import jakarta.ws.rs.core.Link;
21  import jakarta.ws.rs.core.Response;
22  import jakarta.ws.rs.core.UriInfo;
23  
24  import jaxrs.examples.link.clusterservice.Cluster.Status;
25  
26  /**
27   * ClusterResource class.
28   *
29   * @author Santiago.Pericas-Geertsen@oracle.com
30   */
31  @Path("/cluster")
32  public class ClusterResource {
33  
34      @Context
35      private UriInfo uriInfo;
36  
37      private Cluster cluster = Model.getCluster();
38  
39      @GET
40      @Produces({ "application/json" })
41      public Response self() {
42          return Response.ok(cluster).links(getTransitionalLinks()).build();
43      }
44  
45      @POST
46      @Path("onliner")
47      @Produces({ "application/json" })
48      public Response onliner() {
49          cluster.setStatus(Status.ONLINE);
50          return Response.ok(cluster).links(getTransitionalLinks()).build();
51      }
52  
53      @POST
54      @Path("offliner")
55      @Produces({ "application/json" })
56      public Response offliner() {
57          cluster.setStatus(Status.OFFLINE);
58          return Response.ok(cluster).links(getTransitionalLinks()).build();
59      }
60  
61      private Link[] getTransitionalLinks() {
62          URI uri = uriInfo.getRequestUri();
63          URI baseUri = uriInfo.getBaseUri();
64          Link self = Link.fromMethod(getClass(), "self").baseUri(baseUri).rel("self").buildRelativized(uri);
65          Link item = Link.fromMethod(MachineResource.class, "self").baseUri(baseUri).rel("item").buildRelativized(uri);
66          Link onliner = Link.fromMethod(getClass(), "onliner").baseUri(baseUri).rel("onliner").buildRelativized(uri);
67          Link offliner = Link.fromMethod(getClass(), "offliner").baseUri(baseUri).rel("offliner").buildRelativized(uri);
68  
69          return cluster.getStatus() == Status.ONLINE
70                  ? new Link[] {self, item, offliner}
71                  : new Link[] {self, item, onliner};
72      }
73  
74  }
