8   package jaxrs.examples.multipart;
9   
10  import java.io.IOException;
11  import java.io.InputStream;
12  import java.nio.file.Files;
13  import java.nio.file.Path;
14  import java.nio.file.Paths;
15  import java.util.List;
16  import java.util.logging.Level;
17  import java.util.logging.Logger;
18  import java.util.stream.Collectors;
19  
20  import jakarta.ws.rs.client.Client;
21  import jakarta.ws.rs.client.ClientBuilder;
22  import jakarta.ws.rs.client.Entity;
23  import jakarta.ws.rs.client.WebTarget;
24  import jakarta.ws.rs.core.EntityPart;
25  import jakarta.ws.rs.core.GenericType;
26  import jakarta.ws.rs.core.MediaType;
27  import jakarta.ws.rs.core.Response;
28  
29  
30  public class MultipartClient {
31      private static final Logger LOG = Logger.getLogger(MultipartClient.class.getName());
32  
33      public boolean sendPdfs(Path dir) throws IOException {
34          List<EntityPart> parts = Files.list(dir).map(this::toPart).collect(Collectors.toList());
35          Client client = ClientBuilder.newClient();
36          WebTarget target = client.target("http://localhost:9080/multipart?dirName=abc");
37          Entity<List<EntityPart>> entity = Entity.entity(parts, MediaType.MULTIPART_FORM_DATA);
38          Response response = target.request().post(entity);
39          return response.getStatus() == 200;
40      }
41  
42      private EntityPart toPart(Path file) {
43          String filename = file.getFileName().toString();
44          try {
45              return EntityPart.withName(filename)
46                               .content(filename, Files.newInputStream(file))
47                               .mediaType("application/pdf")
48                               .build();
49          } catch (IOException ioex) {
50              LOG.log(Level.WARNING, "Failed to process file {0}", file);
51              return null;
52          }
53      }
54  
55      public List<Path> retrievePdfs(String remoteDirName) throws IOException {
56          Client client = ClientBuilder.newClient();
57          WebTarget target = client.target("http://localhost:9080/multipart").queryParam("dirName", remoteDirName);
58          Response response = target.request(MediaType.MULTIPART_FORM_DATA).get();
59          List<EntityPart> parts = response.readEntity(new GenericType<List<EntityPart>>() { });
60          return parts.stream().map(part -> {
61              try (InputStream is = part.getContent()) {
62                  Path file = Files.createFile(Paths.get(part.getFileName().orElse(part.getName() + ".pdf")));
63                  Files.copy(is, file);
64                  return file;
65              } catch (IOException ioex) {
66                  LOG.log(Level.WARNING, "Failed to process attachment part {0}", part);
67                  return null;
68              }
69          }).collect(Collectors.toList());
70      }
71  }
