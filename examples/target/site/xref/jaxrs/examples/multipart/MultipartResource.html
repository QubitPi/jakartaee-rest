33  @Path("/multipart")
34  public class MultipartResource {
35  
36      private static final String PDF_ROOT_DIR = System.getProperty("pdf.root.dir", "/myPDFs");
37  
38      @GET
39      @Produces(MediaType.MULTIPART_FORM_DATA)
40      public List<EntityPart> getAllPdfFilesInDirectory(@QueryParam("dirName") String dirName) throws IOException {
41          File dir = getDirectoryIfExists(dirName);
42          List<EntityPart> parts = new ArrayList<>();
43          for (File f : dir.listFiles()) {
44              parts.add(EntityPart.withFileName(f.getName()).content(new FileInputStream(f))
45                                                            .mediaType("application/pdf")
46                                                            .build());
47          }
48          return parts;
49      }
50  
51      @POST
52      @Consumes(MediaType.MULTIPART_FORM_DATA)
53      public Response postNewPdfFiles(@QueryParam("dirName") String dirName, List<EntityPart> parts) throws IOException {
54          File dir = getDirectoryIfExists(dirName);
55          for (EntityPart p : parts) {
56              File f = new File(dir, p.getFileName().orElseThrow(BadRequestException::new));
57              if (f.exists()) {
58                  throw new WebApplicationException(409); // 409 CONFLICT
59              }
60              try (InputStream content = p.getContent()) {
61                  Files.copy(content, f.toPath());
62              }
63          }
64          return Response.ok().build();
65      }
